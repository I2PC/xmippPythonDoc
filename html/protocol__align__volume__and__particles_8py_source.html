151  def alignVolumeStep(self, maskArgs):
152 
153  fhInputTranMat = self.getTransformationFile()
154  outVolFn = self.getOutputAlignedVolumePath()
155 
156  args = "--i1 %s --i2 %s --apply %s" % \

197  transform = Transform()
