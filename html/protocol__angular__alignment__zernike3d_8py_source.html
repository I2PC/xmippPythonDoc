181  def createOutputStep(self):
182  from sklearn.manifold import TSNE
183  Xdim = self.inputParticles.get().getXDim()
184  self.Ts = self.inputParticles.get().getSamplingRate()
185  newTs = self.targetResolution.get() * 1.0 /3.0
186  self.newTs = max(self.Ts, newTs)
187  self.newXdim = int(Xdim * self.Ts / newTs)
188  fnOut = self._getFileName('fnOut')
189  mdOut = md.MetaData(fnOut)
190  coeffMatrix = np.vstack(mdOut.getColumnValues(md.MDL_SPH_COEFFICIENTS))
191  X_tsne_1d = TSNE(n_components=1).fit_transform(coeffMatrix)
192  X_tsne_2d = TSNE(n_components=2).fit_transform(coeffMatrix)
193 
194  newMdOut = md.MetaData()
195  i = 0
196  for row in md.iterRows(mdOut):
197  newRow = row
198  newRow.setValue(md.MDL_SPH_TSNE_COEFF1D, float(X_tsne_1d[i, 0]))
199  newRow.setValue(md.MDL_SPH_TSNE_COEFF2D, [float(X_tsne_2d[i, 0]), float(X_tsne_2d[i, 1])])
200  if self.newTs != self.Ts:
201  coeffs = mdOut.getValue(md.MDL_SPH_COEFFICIENTS, row.getObjId())
202  correctionFactor = self.inputVolume.get().getDim()[0] / self.newXdim
203  coeffs = [correctionFactor*coeff for coeff in coeffs]
204  newRow.setValue(md.MDL_SPH_COEFFICIENTS, coeffs)
205  newRow.addToMd(newMdOut)
206  i += 1
207  newMdOut.write(fnOut)
