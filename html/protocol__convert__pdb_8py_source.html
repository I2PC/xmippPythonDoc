106  def convertPdbStep(self):
107  """ Although is not mandatory, usually is used by the protocol to
108  register the resulting outputs in the database.
109  """
110  pdbFn = self._getPdbFileName()
111  outFile = removeExt(self._getVolName())
112  if getExt(pdbFn)==".cif":
113  pdbFn2=replaceBaseExt(pdbFn, 'pdb')
114  cifToPdb(pdbFn, pdbFn2)
115  pdbFn = pdbFn2
116 
117  samplingR = self.sampling.get()
118 
119  args = '-i %s --sampling %f -o %s' % (pdbFn, samplingR, outFile)
120 
121  if self.centerPdb:
122  args += ' --centerPDB'
123 
124  if self.vol:
125  vol = self.volObj.get()
126  size = vol.getDim()
127  ccp4header = headers.Ccp4Header(vol.getFileName(), readHeader=True)
128  self.shifts = ccp4header.getOrigin()
129  args += ' --size %d %d %d --orig %d %d %d' % (size[2], size[1], size[0],
130  self.shifts[0]/samplingR,
131  self.shifts[1]/samplingR,
132  self.shifts[2]/samplingR)
133 
134  if self.setSize:
135  args += ' --size'
136 
137  if self.size_x.hasValue():
138  args += ' %d' % self.size_x.get()
139 
140  if self.size_y.hasValue() and self.size_z.hasValue():
141  args += ' %d %d' % (self.size_y.get(), self.size_z.get())
142 
143  self.info("Input file: " + pdbFn)
144  self.info("Output file: " + outFile)
145 
146  program = "xmipp_volume_from_pdb"
147  self.runJob(program, args)
