111  def extractUnit(self):
112  sym = self.symmetryGroup.get()
113  if sym == XMIPP_CYCLIC:
114  sym = "%s%d" % (XMIPP_SYM_NAME[XMIPP_CYCLIC][:1], self.symmetryOrder)
115  elif sym == XMIPP_DIHEDRAL_X:
116  sym = "%s%d" % \
117  (XMIPP_SYM_NAME[XMIPP_DIHEDRAL_X][:1], self.symmetryOrder)
118  elif sym == XMIPP_TETRAHEDRAL:
119  sym = "%s" % XMIPP_SYM_NAME[XMIPP_TETRAHEDRAL]
120  elif sym == XMIPP_OCTAHEDRAL:
121  sym = "%s" % XMIPP_SYM_NAME[XMIPP_OCTAHEDRAL]
122  elif sym >= XMIPP_I222 and sym <= XMIPP_In25r :
123  sym = XMIPP_SYM_NAME[sym]
124 
125  inFileName = self.inputVolumes.get().getFileName()
126  if inFileName.endswith('.mrc'):
127  inFileName = inFileName + ":mrc"
128  args = "-i %s -o %s" % \
129  (inFileName, self._getOutputVol())
130  args += " --unitcell %s " % sym
131  args += " %f " % self.innerRadius.get()
132  args += " %f " % self.outerRadius.get()
133  args += " %f " % self.expandFactor.get()
134  args += " %f " % self.offset.get()
135  sampling = self.inputVolumes.get().getSamplingRate()
136  args += " %f " % sampling
137  origin = self.inputVolumes.get().getShiftsFromOrigin()
138  # x origin coordinate (from Angstroms to pixels)
139  args += " %f " % (origin[0] / (-1. * sampling))
140  # y origin coordinate (from Angstroms to pixels)
141  args += " %f " % (origin[1] / (-1. * sampling))
142  # z origin coordinate (from Angstroms to pixels)
143  args += " %f " % (origin[2] / (-1. * sampling))
144 
145  self.runJob("xmipp_transform_window", args)
