131  def buildSetOfClasses(self, nclust):
132  '''From a list of clusterings, nclust index it and that clustering
133  is converted into a scipion setOfClasses'''
134  clustIdx = self.nclusters.index(nclust)
135  clustering = self._allClusterings[clustIdx]
136 
137  inputParticles = self.protocol.inputMultiClasses[0].get().getImages()
138  outputClasses = self.protocol._createSetOfClasses3D(inputParticles)
139  for classItem in clustering:
140  numOfPart = classItem[0]
141  partIds = classItem[1]
142  setRepId = classItem[2]
143  clsRepId = classItem[3]
144 
145  setRep = self.protocol.inputMultiClasses[setRepId].get()
146  clRep = setRep[clsRepId]
147 
148  newClass = Class3D()
149  # newClass.copyInfo(clRep)
150  newClass.setAcquisition(clRep.getAcquisition())
151  newClass.setRepresentative(clRep.getRepresentative())
152 
153  outputClasses.append(newClass)
154 
155  enabledClass = outputClasses[newClass.getObjId()]
156  enabledClass.enableAppend()
157  for itemId in partIds:
158  enabledClass.append(inputParticles[itemId])
159  outputClasses.update(enabledClass)
160 
161  return outputClasses
