335  def substractBlocresStep(self):
336 
337  params = ' -i %s' % self._getFileName(BLOCRES_AVG_FILE)+':mrc'
338  params += ' --minus %s' % self._getFileName(BLOCRES_HALF_FILE)+':mrc'
339  params += ' -o %s ' % self._getFileName(RESTA_FILE)
340  self.runJob('xmipp_image_operate', params)
341 
342  Ccp4Header.fixFile(self._getFileName(RESTA_FILE), self._getFileName(RESTA_FILE_MRC),
343  self.origin, self.sampling, Ccp4Header.START)
344 
345  """Diveded by resolution"""
346  Vx = xmipp3.Image(self._getFileName(RESTA_FILE))
347  V=Vx.getData()
348  Vmask = xmipp3.Image(self._getFileName(MASK_FILE_MRC)+':mrc').getData()
349  Vres = xmipp3.Image(self._getFileName(BLOCRES_HALF_FILE)+':mrc').getData()
350  Vt = V
351  Zdim, Ydim, Xdim = V.shape
352 
353  for z in range(0,Zdim):
354  for y in range(0,Ydim):
355  for x in range(0,Xdim):
356  if (Vmask[z,y,x] > 0.001 and Vres[z,y,x]>0.001):
357  Vt[z,y,x] = (V[z,y,x]/Vres[z,y,x])
358  Vx.setData(Vt)
359  Vx.write(self._getFileName(RESTA_FILE_NORM))
360  Ccp4Header.fixFile(self._getFileName(RESTA_FILE_NORM), self._getFileName(RESTA_FILE_NORM),
361  self.origin, self.sampling, Ccp4Header.START)

387  def _setMetrics(self):
388  """ Internal method to compute some metrics. """
389  # mean values of FSC-Q
390 
391  mtd = md.MetaData()
392  mtd.read(self._getFileName(MD_MEANS))
393 
394  mean = mtd.getValue(MDL_VOLUME_SCORE1, 1)
395  meanA = mtd.getValue(MDL_VOLUME_SCORE2, 1)
396 
397  # means value for map divided by resolution (FSC-Qr)
398  mtd2 = md.MetaData()
399  mtd2.read(self._getFileName(MD2_MEANS))
400 
401  mean2 = mtd2.getValue(MDL_VOLUME_SCORE1, 1)
402  meanA2 = mtd2.getValue(MDL_VOLUME_SCORE2, 1)
403 
404  # statistic from fnal pdb with fsc-q
405  # Number of atoms greater or less than 0.5
406  total_atom = 0
407  fscq_greater = 0
408  fscq_less = 0
409  with open(self._getFileName(PDB_VALUE_FILE)) as f:
410  lines_data = f.readlines()
411  for j, lin in enumerate(lines_data):
412 
413  if (lin.startswith('ATOM') or lin.startswith('HETATM')):
414 
415  total_atom = total_atom + 1
416  fscq_atom = float(lin[54:60])
417 
418  if (fscq_atom > 0.5):
419  fscq_greater = fscq_greater + 1
420 
421  if (fscq_atom < -0.5):
422  fscq_less = fscq_less + 1
423 
424  porc_greater = (fscq_greater * 100) / total_atom
425  porc_less = (fscq_less * 100) / total_atom
426 
427  self.mean = Float(mean)
428  self.meanA = Float(meanA)
429  self.mean2 = Float(mean2)
430  self.meanA2 = Float(meanA2)
431  self.total_atom = Integer(total_atom)
432  self.fscq_greater = Integer(fscq_greater)
433  self.fscq_less = Integer(fscq_less)
434  self.porc_greater = Float(porc_greater)
435  self.porc_less = Float(porc_less)
436  self._store()

439  fscqDic = {}

445  resId = '{}:{}@{}'.format(resChain, resNumber, atomName)
