47 def mds(d, dimensions=2):
48  """
49  Multidimensional Scaling - Given a matrix of interpoint distances,
50  find a set of low dimensional points that have similar interpoint
51  distances.
52  """
53 
54  # Distance matrix size
55  (n, n) = d.shape
56 
57  # Centering Matrix
58  J = np.identity(n, float) - (1 / n) * np.ones([n, n], float)
59 
60  # Center distance matrix
61  B = -0.5 * J @ d @ J
62 
63  # Singular value decomposition
64  [U, S, V] = np.linalg.svd(B)
65  S = np.diag(S)
66 
67  # Coordinates matrix from MDS
68  Y = U[:, 0:dimensions] @ np.power(S[0:dimensions, 0:dimensions], 0.5)
69 
70  return Y

72 def rigidRegistration(X, Y):
73  Xcm = np.sum(X, axis=0) / X.shape[0]
74  Ycm = np.sum(Y, axis=0) / Y.shape[0]
75  Xc = np.transpose(X - Xcm)
76  Yc = np.transpose(Y - Ycm)
77  [U, S, V] = np.linalg.svd(Xc @ Yc.T)
78  R = V @ U.T
79  t = Ycm - R @ Xcm
80  return R, t

335  def rigidAlignmentMapping(self, i):
336  X1 = np.loadtxt(self._defineResultsName(i))
337  X2 = np.loadtxt(self._defineResultsName2(i))
338 
339  # Normalize the matrices
340  [_, S, _] = np.linalg.svd(X1)
341  X1 = X1 / np.amax(np.abs(S))
342  [_, S, _] = np.linalg.svd(X2)
343  X2 = X2 / np.amax(np.abs(S))
344 
345  # Register the points (taking mirrors into account)
346  for idf in range(4):
347  if idf == 0:
348  R, t = rigidRegistration(X1, X2)
349  X1 = t + np.transpose(R @ X1.T)
350  cost = np.sum(pdist(X1 - X2))
351  elif idf == 1:
352  X1_attempt = np.fliplr(X1)
353  R, t = rigidRegistration(X1_attempt, X2)
354  X1_attempt = t + np.transpose(R @ X1_attempt.T)
355  cost_attempt = np.sum(pdist(X1_attempt - X2))
356  if cost_attempt < cost:
357  cost = cost_attempt
358  X1 = X1_attempt
359  elif idf == 2:
360  X1_attempt = np.flipud(X1)
361  R, t = rigidRegistration(X1_attempt, X2)
362  X1_attempt = t + np.transpose(R @ X1_attempt.T)
363  cost_attempt = np.sum(pdist(X1_attempt - X2))
364  if cost_attempt < cost:
365  cost = cost_attempt
366  X1 = X1_attempt
367  elif idf == 3:
368  X1_attempt = np.fliplr(X1)
369  X1_attempt = np.flipud(X1_attempt)
370  R, t = rigidRegistration(X1_attempt, X2)
371  X1_attempt = t + np.transpose(R @ X1_attempt.T)
372  cost_attempt = np.sum(pdist(X1_attempt - X2))
373  if cost_attempt < cost:
374  cost = cost_attempt
375  X1 = X1_attempt

393  def convolution(self, Xr, kernel, sigma, R, C, D):
394  # Create the grid
395  lbox = int(6 * sigma)
396  if lbox % 2 == 0:
397  lbox += 1
398  mid = int((lbox - 1) / 2 + 1)
399  S_shape = [d + lbox for d in R.shape]
400  S = np.zeros(S_shape)

404  if D is None:

406  indx = np.argmin(np.abs(C[0, :] - Xr[p, 1]))
407  S[indx:indx + 2 * mid - 1, indy:indy + 2 * mid - 1] += kernel
408  else:
409  indx = np.argmin(np.abs(R[:, 0, 0] - Xr[p, 0]))
410  indy = np.argmin(np.abs(C[0, :, 0] - Xr[p, 1]))
411  indz = np.argmin(np.abs(D[0, 0, :] - Xr[p, 2]))
412  S[indx:indx + 2 * mid - 1, indy:indy + 2 * mid - 1, indz:indz + 2 * mid - 1] += kernel

415  def entropyConsensus(self):
416  for i in range(2, 4):
417  # Rigid alignment of mappings (considering mirrors as well)
418  X1, X2 = self.rigidAlignmentMapping(i)
419 
420  # Round point to place them in a grid
421  Xr1 = np.round(X1, decimals=3)
422  Xr2 = np.round(X2, decimals=3)
423  size_grid = 2.5 * max((np.amax(Xr1), np.amax(Xr2)))
424 
425  # Parameters needed for future convolution
426  grid_coords = np.linspace(-size_grid, size_grid, num=400)
427  if i == 2:
428  R, C = np.meshgrid(grid_coords, grid_coords, indexing='ij')
429  D = None
430  else:
431  R, C, D = np.meshgrid(grid_coords, grid_coords, grid_coords, indexing='ij')
432  sigma = R.shape[0] / 20
433 
434  # Create Gaussian Kernel
435  kernel, lbox, mid = self.gaussianKernel(sigma, i)
436 
437  # Consensus
438  alpha_vect = np.arange(0, 1.01, 0.01)
439  entropy_vect = []
440  X_matrices = []
441 
442  for alpha in alpha_vect:
443  X = alpha * X1 + (1 - alpha) * X2
444  X_matrices.append(X)
445 
446  # Round points to place them in a grid
447  Xr = np.round(X, decimals=3)
448 
449  # Convolve grid and kernel
450  S = self.convolution(Xr, kernel, sigma, R, C, D)
451 
452  # Compute the Shannon entropy associated to the convolved grid
453  _, counts = np.unique(S, return_counts=True)
454  entropy_vect.append(entropy(counts, base=2))

465  id_optimal = 0
