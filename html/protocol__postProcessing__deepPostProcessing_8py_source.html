202  def deepVolPostProStep(self):
203  outputFname= self._getExtraPath(POSTPROCESS_VOL_BASENAME)
204  if os.path.isfile(outputFname):
205  return
206 
207  if self.useHalfMapsInsteadVol.get():
208  half1= self._getTmpPath(INPUT_HALF1_BASENAME)
209  half2= self._getTmpPath(INPUT_HALF2_BASENAME)
210  params=" -i %s -i2 %s"%(half1, half2)
211  else:
212  inputFname = self._getTmpPath(INPUT_VOL_BASENAME)
213  params=" -i %s "%inputFname
214 
215  params+=" -o %s "%outputFname
216  params+= " --sampling_rate %f "%(self.inputVolume.get().getSamplingRate() if self.inputVolume.get() is not None
217  else self.inputHalf1.get().getSamplingRate())
218  params+= " -b %s " %(self.batch_size)
219 
220  if self.useQueueForSteps() or self.useQueue():
221  params += ' -g all '
222  else:
223  params += ' -g %s' % (",".join([str(elem) for elem in self.getGpuList()]))
224 
225  if self.normalization==self.NORMALIZATION_MASK:
226  params+= " --binaryMask %s "%(self._getTmpPath(INPUT_MASK_BASENAME))
227  elif self.normalization==self.NORMALIZATION_STATS:
228  params+= " --noise_stats_mean %f --noise_stats_std %f "%(self.noiseMean, self.noiseStd)
229 
230 
231  if self.performCleaningStep:
232  params+= " --cleaningStrengh %f" %self.sizeFraction_CC.get()
233  else:
234  params+= " --cleaningStrengh -1 "
235 
236  if self.normalization in [self.NORMALIZATION_AUTO, self.NORMALIZATION_STATS]:
237  if self.modelType == self.TIGHT_MODEL:
238  params+= " --checkpoint %s "%self.getModel("deepEMhancer", "production_checkpoints/deepEMhancer_tightTarget.hd5")
239  elif self.modelType == self.HI_RES:
240  params+= " --checkpoint %s "%self.getModel("deepEMhancer", "production_checkpoints/deepEMhancer_highRes.hd5")
241  else:
242  params+= " --checkpoint %s "%self.getModel("deepEMhancer", "production_checkpoints/deepEMhancer_wideTarget.hd5")
243  else: #self.NORMALIZATION_MASK
244  params+= " --checkpoint %s "%self.getModel("deepEMhancer", "production_checkpoints/deepEMhancer_masked.hd5")
245 
246  self.runJob("xmipp_deep_volume_postprocessing", params, numberOfMpi=1)
