47 class XmippProtConsensusPicking(ProtParticlePicking):
48  """
49  Protocol to estimate the agreement between different particle picking
50  algorithms. The protocol takes several Sets of Coordinates calculated
51  by different programs and/or different parameter settings. Let's say:
52  we consider N independent pickings. Then, a coordinate is considered
53  to be a correct particle if M pickers have selected the same particle
54  (within a radius in pixels specified in the form).
55 
56  If you want to be very strict, then set M=N; that is, a coordinate
57  represents a particle if it has been selected by all particles (this
58  is the default behaviour). Then you may relax this condition by setting
59  M=N-1, N-2, ...
60 
61  If you want to be very flexible, set M=1, in this way it suffices that
62  1 picker has selected the coordinate to be considered as a particle. Note
63  that in this way, the cleaning of the dataset has to be performed by other
64  means (screen particles, 2D and 3D classification, ...).
65  """

304 def consensusWorker(coords, consensus, consensusRadius, posFn, jaccFn=None,
305  mode=PICK_MODE_LARGER):
306  """ Worker for calculate the consensus of N picking algorithms of
307  M_n coordinates each one.
308 
309  coords: Array of N numpy arrays of M_n coordinates each one.
310  consensus: Minimum number of votes to get a consensus coordinate
311  consensusRadius: Tolerance to see two coordinates as the same (in pixels)
312  posFn: Where to write the consensus coordinates
313  jaccFN: Where to write the Jaccard index per micrograph
314  """
315  if len(coords) == 1: # self consensus (remove duplicates)
316  N0 = 0
317  firstInput = 0
318  else: # in regular consensus all first coords are directly added to allCoords
319  N0 = coords[0].shape[0]
320  firstInput = 1
321 
322  # initializing arrays
323  Ninputs = len(coords)
324  Ncoords = sum([x.shape[0] for x in coords])
325  allCoords = np.zeros([Ncoords, 2])
326  votes = np.zeros(Ncoords)
327 
328  inAllMicrographs = consensus <= 0 or consensus >= Ninputs
329 
330  # if nothing in the first and it should be in all, nothing to do
331  if (not all([coords[idx].shape[0] for idx in range(Ninputs)])
332  and inAllMicrographs):
333  print("Returning from worker: doing AND consensus and, at least, one "
334  "picker is empty for this micrograph (%s)." % posFn)
335  return
336 
337  # Add all the first coordinates to 'allCoords' and 'votes' lists
338  if N0 > 0:
339  allCoords[0:N0, :] = coords[0]
340  votes[0:N0] = 1
341 
342  # Add the rest of coordinates to 'allCoords' and 'votes' lists
343  Ncurrent = N0
344  for n in range(firstInput, Ninputs):
345  for coord in coords[n]:
346  if Ncurrent > 0:
347  dist = np.sum((coord - allCoords[0:Ncurrent]) ** 2, axis=1)
348  imin = np.argmin(dist)
349  if sqrt(dist[imin]) < consensusRadius:
350  newCoord = (votes[imin] * allCoords[imin,] + coord) / (
351  votes[imin] + 1)
352  allCoords[imin,] = newCoord
353  votes[imin] += 1
354  else:
355  allCoords[Ncurrent, :] = coord
356  votes[Ncurrent] = 1
357  Ncurrent += 1
358  else:
359  allCoords[Ncurrent, :] = coord
360  votes[Ncurrent] = 1
361  Ncurrent += 1
362 
363  # Select those in the consensus
364  if consensus <= 0 or consensus > Ninputs:
365  consensus = Ninputs
366  elif not isinstance(consensus, int):
367  consensus = consensus.get()
368 
369  if mode==PICK_MODE_LARGER:
370  consensusCoords = allCoords[votes >= consensus, :]
371  else:
372  consensusCoords = allCoords[votes == consensus, :]
373 
374  try:
375  if jaccFn:
376  jaccardIdx = float(len(consensusCoords)) / (
377  float(len(allCoords)) / Ninputs)
378  # COSS: Possible problem with concurrent writes
379  with open(jaccFn, "a") as fhJaccard:
380  fhJaccard.write("%s \t %f\n" % (posFn, jaccardIdx))
381  except Exception as exc:
382  print("Some error occurred during Jaccard index calculation or "
383  "writing it's file. Maybe a concurrence issue:\n%s" % exc)
384  # Write the consensus file only if there
385  # are some coordinates (size > 0)
386  if consensusCoords.size:
387  np.savetxt(posFn, consensusCoords)
