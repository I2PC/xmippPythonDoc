143  def minimum_spanning_tree(self):
144  N = self.coords.shape[0]
145  tree = KDTree(self.coords)
146  distances, indices = tree.query(self.coords, k=10)
147  nn_mat = np.zeros((N, N))
148  for idn in range(N):
149  nn_mat[idn, indices[idn]] += distances[idn].reshape(-1)
150  nn_mat[idn, idn] = 0
151 
152  edges = ((int(e[0]), int(e[1])) for e in zip(*np.asarray(nn_mat).nonzero()))
153  triples = list(((u, v, float(nn_mat[u, v])) for u, v in edges))
154  edges = [(triple[0], triple[1]) for triple in triples]
155  weigths_edge = [triple[2] for triple in triples]
156 
157  edge_matrix = np.zeros((N, N))
158  for edge, weigth in zip(edges, weigths_edge):
159  edge_matrix[edge] = weigth
160  self.T = self.KruskalMST(triples, N)

162  def KruskalMST(self, triples, N):
163  result = []
164  i = 0
165  e = 0
166  graph = sorted(triples, key=lambda item: item[2])
167  parent = []
168  rank = []
169  for node in range(N):
170  parent.append(node)
171  rank.append(0)
172  while e < N - 1:
173  u, v, w = graph[i]
174  i = i + 1
175  x = self.find(parent, u)
176  y = self.find(parent, v)
177  if x != y:
178  e = e + 1
179  result.append([u, v, w])
180  self.union(parent, rank, x, y)
181  return result

183  def find(self, parent, i):
184  if parent[i] == i:
185  return i
186  return self.find(parent, parent[i])

188  def union(self, parent, rank, x, y):
189  xroot = self.find(parent, x)
190  yroot = self.find(parent, y)
191  if rank[xroot] < rank[yroot]:
192  parent[xroot] = yroot
193  elif rank[xroot] > rank[yroot]:
194  parent[yroot] = xroot
195  else:
196  parent[yroot] = xroot
197  rank[xroot] += 1

274  def plotConvolution(self, x, y):
275  coordinates = np.stack((x, y), axis=1)
276  Xr = np.round(coordinates, decimals=3)
277  size_grid = 2 * np.amax(Xr)
278  grid_coords = np.arange(-size_grid, size_grid, 0.001)
279  R, C = np.meshgrid(grid_coords, grid_coords, indexing='ij')
280  S = np.zeros(R.shape)
281  sigma = R.shape[0] / (200 / 5)
282  lbox = int(6 * sigma)
283  if lbox % 2 == 0:
284  lbox += 1
285  mid = int((lbox - 1) / 2 + 1)
286  kernel = np.zeros((lbox, lbox))
287  kernel[mid, mid] = 1
288  kernel = gaussian_filter(kernel, sigma=sigma)
289  for p in range(Xr.shape[0]):
290  indx = np.argmin(np.abs(R[:, 0] - Xr[p, 0]))
291  indy = np.argmin(np.abs(C[0, :] - Xr[p, 1]))
292  if 'weights' in locals():
293  S[indx - mid:indx + mid - 1, indy - mid:indy + mid - 1] += kernel * self.weights
294  else:
295  S[indx - mid:indx + mid - 1, indy - mid:indy + mid - 1] += kernel
296  S = S[~np.all(S == 0, axis=1)]
297  S = S[:, ~np.all(S == 0, axis=0)]
