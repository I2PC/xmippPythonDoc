102  def convertInputStep(self):
103  fnInputVol = self._getFileName('fnInputVol')
104  fnRefVol = self._getFileName('fnRefVol')
105 
106  XdimI = self.inputVolume.get().getDim()[0]
107  TsI = self.inputVolume.get().getSamplingRate()
108  XdimR = self.refVolume.get().getDim()[0]
109  TsR = self.refVolume.get().getSamplingRate()
110  self.newTs = self.targetResolution.get() * 1.0 / 3.0
111  self.newTs = max(TsI, TsR, self.newTs)
112  newXdimI = XdimI * TsI / self.newTs
113  newXdimR = XdimR * TsR / self.newTs
114  newRmax = self.Rmax.get() * TsI / self.newTs
115  self.newXdim = min(newXdimI, newXdimR)
116  self.newRmax = min(newRmax, self.Rmax.get())
117 
118  ih = ImageHandler()
119  ih.convert(self.inputVolume.get(), fnInputVol)
120  if XdimI != newXdimI:
121  self.runJob("xmipp_image_resize",
122  "-i %s --dim %d" % (fnInputVol, newXdimI))
123  if newXdimI > self.newXdim:
124  self.runJob("xmipp_transform_window", " -i %s --crop %d" %
125  (fnInputVol, (newXdimI-self.newXdim)))
126 
127 
128  ih.convert(self.refVolume.get(), fnRefVol)
129 
130  if XdimR != newXdimR:
131  self.runJob("xmipp_image_resize",
132  "-i %s --dim %d" % (fnRefVol, newXdimR))
133  if newXdimR > self.newXdim:
134  self.runJob("xmipp_transform_window", " -i %s --crop %d" %

162  def createOutputStep(self):
163  if self.targetResolution.get() != 3.0:
164  correctionFactor = self.targetResolution.get() / 3.0
165  with open(self._getExtraPath('Volumes_clnm.txt'), 'r') as fid:
166  lines = fid.readlines()
167  basisParams = np.fromstring(lines[0], sep=' ')
168  if self.Rmax.get() != 0:
169  basisParams[2] = self.Rmax.get()
170  else:
171  basisParams[2] = self.refVolume.get().getDim()[0] / 2
172  clnm = np.fromstring(lines[1], sep=' ') * correctionFactor
173  with open(self._getExtraPath('Volumes_clnm.txt'), 'w') as fid:
174  fid.write(' '.join(map(str, basisParams)) + "\n")
175  fid.write(' '.join(map(str, clnm)) + "\n")
176  vol = Volume()

178  vol.setSamplingRate(self.newTs)

182  def alignMaps(self):
183  fnInputVol = self._getFileName('fnInputVol')
184  fnInputFilt = self._getFileName('fnInputFilt')
185  fnRefVol = self._getFileName('fnRefVol')
186  fnRefFilt = self._getFileName('fnRefFilt')
187  fnOutVol = self._getFileName('fnOutVol')
188 
189  # Filter the volumes to improve alignment quality
190  params = " -i %s -o %s --fourier real_gaussian 2" % (fnInputVol, fnInputFilt)
191  self.runJob("xmipp_transform_filter", params)
192  params = " -i %s -o %s --fourier real_gaussian 2" % (fnRefVol, fnRefFilt)
193  self.runJob("xmipp_transform_filter", params)
194 
195  # Find transformation needed to align the volumes
196  params = ' --i1 %s --i2 %s --local --dontScale ' \
197  '--copyGeo %s' % \
198  (fnRefFilt, fnInputFilt, self._getExtraPath("geo.txt"))
199  self.runJob("xmipp_volume_align", params)
200 
201  # Apply transformation of filtered volume to original volume
202  with open(self._getExtraPath("geo.txt"), 'r') as file:
203  geo_str = file.read().replace('\n', ',')
204  params = " -i %s -o %s --matrix %s" % (fnInputVol, fnOutVol, geo_str)
205  self.runJob("xmipp_transform_geometry", params)
