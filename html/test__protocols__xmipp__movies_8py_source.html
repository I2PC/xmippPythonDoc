707  def _checkMaxShiftFiltering(self, protocol, label, hasMic, hasDw=False, results=[]):

713  def assertOutput(outputName, ids=[1, 2]):
714  """ Check if outputName exists and if so, if it's right. (for each id)
715  """
716  print("checking '%s' in movies %s" % (outputName, ids))
717  targetSamplingRate = protocol.inputMovies.get().getSamplingRate()
718  if 'Micrographs' in outputName:
719  # The Mics has n=1
720  if hasDw:
721  inputDim = (1950, 1950, 1)
722  else:
723  # The corr. align. prot. crops the micrographs and is binned
724  inputDim = (970, 967, 1)
725  targetSamplingRate *= 2
726  else:
727  inputDim = protocol.inputMovies.get().getDim()
728 
729  DWstr = ' (DoseWeighted)' if hasDw else ''
730  for itemId in ids:
731  output = getattr(protocol, outputName, None)

733  "Bad filtering in %s test%s."

739  "The size of the movies/mics has changed "
740  "for %s test%s." % (label, DWstr))
741  self.assertEqual(output[itemId].getSamplingRate(),
742  targetSamplingRate,
743  "The samplig rate is incorrect for %s test%s."
744  % (label, DWstr))
745 
746  if all(results):
747  # Checking if only the accepted set is created and
748  # its items have the good size and sampling rate
749  assertOutput('outputMovies')
750  if hasMic:
751  assertOutput('outputMicrographs')
752  if hasDw:
753  assertOutput('outputMicrographsDoseWeighted')
754 
755  # Checking if the Movies/MicsDiscarded set are not created
756  self.assertIsNone(getattr(protocol, 'outputMoviesDiscarded', None),

758  "Bad filtering in %s test." % label)

764  # Checking if only the discarded set is crated and

768  assertOutput('outputMicrographsDiscarded')

770  assertOutput('outputMicrographsDoseWeightedDiscarded')
771 
772  # Checking if the Movie (accepted) set is not created

778  "outputMicrographs (accepted)t were created. "
