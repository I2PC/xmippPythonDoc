412 def rowToAlignmentEmtable(alignmentRow, alignType):
413  """
414  is2D == True-> matrix is 2D (2D images alignment)
415  otherwise matrix is 3D (3D volume alignment or projection)
416  invTransform == True -> for xmipp implies projection
417  """
418 
419  is2D = alignType == ALIGN_2D
420  inverseTransform = alignType == ALIGN_PROJ
421 
422  if alignmentRow.hasAnyColumn(ALIGNMENT_DICT.values()):
423  alignment = Transform()
424  angles = np.zeros(3)
425  shifts = np.zeros(3)
426  flip = alignmentRow.get(XMIPPCOLUMNS.flip.value, default=0.)
427 
428  shifts[0] = alignmentRow.get(XMIPPCOLUMNS.shiftX.value, default=0.)
429  shifts[1] = alignmentRow.get(XMIPPCOLUMNS.shiftY.value, default=0.)
430 
431  if not is2D:
432  angles[0] = alignmentRow.get(XMIPPCOLUMNS.angleRot.value, default=0.)
433  angles[1] = alignmentRow.get(XMIPPCOLUMNS.angleTilt.value, default=0.)
434  angles[2] = alignmentRow.get(XMIPPCOLUMNS.anglePsi.value, default=0.)
435  shifts[2] = alignmentRow.get(XMIPPCOLUMNS.shiftZ.value, default=0.)
436  if flip:
437  angles[1] = angles[1] + 180 # tilt + 180
438  angles[2] = - angles[2] # - psi, COSS: this is mirroring X
439  shifts[0] = - shifts[0] # -x
440  else:
441  psi = alignmentRow.get(XMIPPCOLUMNS.anglePsi.value, default=0.)
442  rot = alignmentRow.get(XMIPPCOLUMNS.angleRot.value, default=0.)
443  if not np.isclose(rot, 0., atol=1e-6 ) and not np.isclose(psi, 0., atol=1e-6):
444  print("HORROR rot and psi are different from zero in 2D case")
445 
446  angles[0] = psi + rot
447 
448  M = matrixFromGeometry(shifts, angles, inverseTransform)
449 
450  if flip:
451  if alignType == ALIGN_2D:
452  M[0, :2] *= -1. # invert only the first two columns
453  # keep x
454  M[2, 2] = -1. # set 3D rot
455  elif alignType == ALIGN_3D:
456  M[0, :3] *= -1. # now, invert first line excluding x
457  M[3, 3] *= -1.
458  elif alignType == ALIGN_PROJ:
459  pass
460 
461  alignment.setMatrix(M)
462 
463  else:
464  alignment = None
465 
466  return alignment
