45 class XmippProtValidateOverfitting(ProtReconstruct3D):
46  """
47  Check how the resolution changes with the number of projections used for
48  3D reconstruction.
49 
50  NOTE:
51  Using the output plot, with the reconstruction of aligned gaussian noise,
52  you can assess the validity of the reconstruction from your micrograph
53  images. Practically, if the resolution of reconstruction based on your
54  images is not considerably different from aligned gaussian noise one
55  (for less number of particles),your images may not produce a valid
56  reconstruction.
57 
58  This method has been proposed by:
59  B. Heymann "Validation of 3D EM Reconstructions", 2015.
60  (see References)
61  """
62  _label = 'validate overfitting'
63  _lastUpdateVersion = VERSION_1_1

477  fnFreqs = sorted(glob.glob(self._getExtraPath(pattern)))

488  for line in fnFreqOpen:
489  fields = line.split()
490  rowdata = list(map(float, fields))
491  val = 1.0/(float(rowdata[0])*float(rowdata[0]))
492  dataInv.append(val)
493  data.extend(rowdata)

496  varRes = (sum(data) / (len(data) - 1))

502  stdResInv = sqrt(varResInv)
