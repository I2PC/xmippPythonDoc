172  def _showVolumesChimera(self):
173  """ Create a chimera script to visualize selected volumes. """
174  volumes = self._getVolumeNames()
175 
176  if len(volumes) > 1:
177  cmdFile = self.protocol._getExtraPath('chimera_volumes.cmd')
178  f = open(cmdFile, 'w+')
179  for volFn in volumes:
180  # We assume that the chimera script will be generated
181  # in the same folder as xmipp volumes
182  localVol = os.path.basename(volFn)
183  if exists(volFn):
184  f.write("open %s\n" % localVol)
185  f.write('tile\n')
186  f.close()
187  view = ChimeraView(cmdFile)
188  else:
189  view = ChimeraClientView(volumes[0])

197  md.activateMathExtensions()

205  blockName = 'class_%06d@' % ref3d
206  for it in self._iterations:
207  fscFn = self.protocol._getFileName('fsc_it', iter=it)
208  if exists(fscFn):
209  fsc = self._plotFSC(None, blockName + fscFn, 'iter %d, class %d' % (it, ref3d))
210  fscSet.append(fsc)
211  fscViewer.visualize(fscSet)
