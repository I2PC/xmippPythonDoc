103  def _insertItersSteps(self):
104  """ Insert several steps needed per iteration. """
105 
106  for iterN in self.allIters():
107  dirsStep = self._insertFunctionStep('createIterDirsStep', iterN)
108  # Insert some steps per reference volume
109  projMatchSteps = []
110  for refN in self.allRefs():
111  # Mask the references in the iteration
112  insertMaskReferenceStep(self, iterN, refN, prerequisites=[dirsStep])
113  # Create the library of projections
114  insertAngularProjectLibraryStep(self, iterN, refN)
115  # Projection matching steps
116  projMatchStep = self._insertProjectionMatchingStep(iterN, refN)
117  projMatchSteps.append(projMatchStep)
118 
119  # Select the reference that best fits each image
120  self._insertFunctionStep('assignImagesToReferencesStep', iterN, prerequisites=projMatchSteps)
121 
122  insertAngularClassAverageStep(self, iterN, refN)
123 
124  # Reconstruct each reference with new averages
125  for refN in self.allRefs():
126  # Create new class averages with images assigned
127  insertReconstructionStep(self, iterN, refN)
128 
129  if self.doComputeResolution and self._doSplitReferenceImages[iterN]:
130  # Reconstruct two halves of the data
131  insertReconstructionStep(self, iterN, refN, 'Split1')

318  def calculateDeviationsStep(self, it):
319  """ Calculate both angles and shifts devitations for all iterations
320  """
321 
322  SL = emlib.SymList()
323  mdIter = emlib.MetaData()
324  #for it in self.allIters():
325  mdIter.clear()
326  SL.readSymmetryFile(self._symmetry[it])
327  md1 = emlib.MetaData(self.docFileInputAngles[it])
328  md2 = emlib.MetaData(self.docFileInputAngles[it-1])
329  #ignore disabled,
330  md1.removeDisabled()
331  md2.removeDisabled()
332 
333  #first metadata file may not have shiftx and shifty
334  if not md2.containsLabel(emlib.MDL_SHIFT_X):
335  md2.addLabel(emlib.MDL_SHIFT_X)
336  md2.addLabel(emlib.MDL_SHIFT_Y)
337  md2.fillConstant(emlib.MDL_SHIFT_X,0.)
338  md2.fillConstant(emlib.MDL_SHIFT_Y,0.)
339  oldLabels=[emlib.MDL_ANGLE_ROT,
340  emlib.MDL_ANGLE_TILT,
341  emlib.MDL_ANGLE_PSI,
342  emlib.MDL_SHIFT_X,
343  emlib.MDL_SHIFT_Y]
344  newLabels=[emlib.MDL_ANGLE_ROT2,
345  emlib.MDL_ANGLE_TILT2,
346  emlib.MDL_ANGLE_PSI2,
347  emlib.MDL_SHIFT_X2,
348  emlib.MDL_SHIFT_Y2]
349  md2.renameColumn(oldLabels,newLabels)
350  md2.addLabel(emlib.MDL_SHIFT_X_DIFF)
351  md2.addLabel(emlib.MDL_SHIFT_Y_DIFF)
352  md2.addLabel(emlib.MDL_SHIFT_DIFF)
353  mdIter.join1(md1, md2, emlib.MDL_IMAGE, emlib.INNER_JOIN)
354  SL.computeDistance(mdIter,False,False,False)
355  emlib.activateMathExtensions()
356  #operate in sqlite
357  shiftXLabel = emlib.label2Str(emlib.MDL_SHIFT_X)
358  shiftX2Label = emlib.label2Str(emlib.MDL_SHIFT_X2)
359  shiftXDiff = emlib.label2Str(emlib.MDL_SHIFT_X_DIFF)
360  shiftYLabel = emlib.label2Str(emlib.MDL_SHIFT_Y)
361  shiftY2Label = emlib.label2Str(emlib.MDL_SHIFT_Y2)
362  shiftYDiff = emlib.label2Str(emlib.MDL_SHIFT_Y_DIFF)
363  shiftDiff = emlib.label2Str(emlib.MDL_SHIFT_DIFF)
364  #timeStr = str(dtBegin)
365  operateString = shiftXDiff+"="+shiftXLabel+"-"+shiftX2Label
366  operateString += "," + shiftYDiff+"="+shiftYLabel+"-"+shiftY2Label
367  mdIter.operate(operateString)
368  operateString = shiftDiff+"=sqrt(" \
369  +shiftXDiff+"*"+shiftXDiff+"+" \
370  +shiftYDiff+"*"+shiftYDiff+");"
371  mdIter.operate(operateString)
372  iterFile = self._mdDevitationsFn(it)
373  mdIter.write(iterFile,emlib.MD_APPEND)
374 
375  self._setLastIter(it)
