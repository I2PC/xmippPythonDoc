280  def add(x): summary.append(' ' + x) # short notation
281  if protocol.filterSpace == FILTER_SPACE_FOURIER:

286  lowAN = Ts/lowA
287  highAN = Ts/highA
288  lowN, highN = protocol.lowFreqDig.get(), protocol.highFreqDig.get()

292  decayStr = ('*%g* A' % protocol.freqDecayA.get() if inA else
293  '*%g*' % protocol.freqDecayDig.get())
294  if mode == cls.FM_LOW_PASS:
295  add('Mode: *Low-pass* ( %s, decay at %s )' %
296  ('resolution > *%g* A (normalized freq. %g)' % (highA,highAN) if inA else
297  'normalized frequency < *%g* (%g A)' % (highN,highNA), decayStr))
298  elif mode == cls.FM_HIGH_PASS:
299  add('Mode: *High-pass* ( %s, decay at %s )' %
300  ('resolution < *%g* A (normalized freq. %g)' % (lowA,lowAN) if inA else
301  'normalized frequency > *%g* (%g A)' % (lowN,lowNA), decayStr))
302  elif mode == cls.FM_BAND_PASS:
303  add('Mode: *Band-pass* ( %s, decay at %s )' %
304  ('*%g* A (normalized freq. %g) < resolution < *%g* A (normalized freq. %g)' % (highA, highAN, lowA, lowAN) if inA else
305  '*%g* (%g A) < normalized frequency < *%g* (%g A)' % (lowN, lowNA, highN, highNA),
306  decayStr))
307  elif protocol.filterSpace == FILTER_SPACE_REAL:
308  add('Mode: *Median*')
309  elif protocol.filterSpace == FILTER_SPACE_WAVELET:
310  add('Space: *Wavelet*')
311  filterMode = protocol.filterModeWavelets.get()
312  if filterMode == cls.FM_DAUB4: add('Wavelet: *Daubechies 4*')
