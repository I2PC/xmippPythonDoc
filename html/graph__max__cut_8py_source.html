35 def maximum_cut_sdp(graph: scipy.sparse.csr_matrix,
36  solver: str = 'SCS' ) -> Tuple[float, Tuple[Set[int], Set[int]]]:
37  n_vertices = graph.shape[0]
38 
39  # Remove the symmetric part of the graph so
40  # that edges are only considrered once
41  graph = scipy.sparse.tril(graph, format='csr')
42 
43  # Obtain the edges and their weights
44  edges = graph.nonzero()
45  weights = graph[edges].A1
46 
47  # Define the objective function and constraints
48  P = cp.Variable(graph.shape, symmetric=True)
49  constraints = [P >> 0, cp.diag(P) == 1] # SDP and diagonal of 1s
50  objective_function = cp.scalar_product(weights, 1.0-P[edges])
51 
52  # Optimize
53  problem = cp.Problem(cp.Maximize(objective_function), constraints)
54  value = problem.solve(solver)
55 
56  # Quantize the result
57  generator = np.random.default_rng()
58  v = scipy.linalg.sqrtm(P.value)
59  u = generator.normal(size=n_vertices)
60  projections = v @ u
61 
62  # Generate the output
63  v0 = set(np.nonzero(projections > 0)[0])
64  v1 = set(range(n_vertices)) - v0 # Complement
65 
66  return (value, (v0, v1))
