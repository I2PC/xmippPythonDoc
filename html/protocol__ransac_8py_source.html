46 class XmippProtRansac(ProtInitialVolume):
47  """
48  Computes an initial 3d model from a set of projections/classes
49  using RANSAC algorithm.
50 
51  This method is based on an initial non-lineal dimensionality
52  reduction approach which allows to select representative small
53  sets of class average images capturing the most of the structural
54  information of the particle under study. These reduced sets are
55  then used to generate volumes from random orientation assignments.
56  The best volume is determined from these guesses using a random
57  sample consensus (RANSAC) approach.
58  """
59  _label = 'ransac'

417  def scoreFinalVolumes(self):
418  threshold=self.getCCThreshold()
419  mdOut=emlib.MetaData()
420  for n in range(self.numVolumes.get()):
421  fnRoot=self._getPath('proposedVolume%05d'%n)

425  MDassignment=emlib.MetaData(fnAssignment)
426  sum=0
427  thresholdedSum=0
428  N=0
429  minCC=2
430  for id in MDassignment:
431  cc=MDassignment.getValue(emlib.MDL_MAXCC,id)
432  sum+=cc
433  thresholdedSum+=cc-threshold
434  if cc<minCC:
435  minCC=cc
436  N+=1
437  if N>0:
438  avg=sum/N
439  else:
440  avg=0.0
441  id=mdOut.addObject()
442  mdOut.setValue(emlib.MDL_IMAGE,fnRoot+".mrc",id)
443  mdOut.setValue(emlib.MDL_VOLUME_SCORE_SUM,float(sum),id)
444  mdOut.setValue(emlib.MDL_VOLUME_SCORE_SUM_TH,float(thresholdedSum),id)
445  mdOut.setValue(emlib.MDL_VOLUME_SCORE_MEAN,float(avg),id)
446  mdOut.setValue(emlib.MDL_VOLUME_SCORE_MIN,float(minCC),id)
