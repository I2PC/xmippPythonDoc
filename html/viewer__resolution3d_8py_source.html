109  def _adjustPoints(self, data):
110  x=data.getXData()
111  if x[0]>x[1]:
112  aux=x[1]
113  x[1]=x[0]
114  x[0]=aux
115  X=[]
116  Y=[]
117  for objId in self.md:
118  f=self.md.getValue(emlib.MDL_RESOLUTION_FREQ2,objId)
119  if f>=x[0] and f<=x[1]:
120  X.append(f)
121  Y.append(self.md.getValue(emlib.MDL_RESOLUTION_LOG_STRUCTURE_FACTOR,objId))
122  X=np.array(X)
123  Y=np.array(Y)
124  A=np.array([np.ones(X.size), X.T])
125  beta=np.linalg.lstsq(A.T,Y)[0]
126  y = [beta[0]+beta[1]*xi for xi in x]
127  Bfactor = -4*beta[1]
128 
129  # Update data
130  data.getPoint(0).setX(x[0])
131  data.getPoint(0).setY(y[0])
132  data.getPoint(1).setX(x[1])
133  data.getPoint(1).setY(y[1])
134  data.bfactor=Bfactor
135 
136  f = open(self.protocol._getPath('bfactor.txt'), 'w')
137  print >> f, x[0], y[0], x[1], y[1], Bfactor
138  f.close()

179  def _applyBfactor(self, e=None):

183  f.close()

260  def onClick(self, event):
261  if event.inaxes!=self.ax:
262  return
263 
264  ex = event.xdata
265  ey = event.ydata
266 
267  if self.drawing == STATE_DRAW_POINTS:
268  point = self.pathData.createEmptyPoint()
269  point.setX(ex)
270  point.setY(ey)
271  self.pathData.addPoint(point)
272 
273  if self.pathData.getSize() == 1: # first point is added
274  self.plotPath()
275  else:
276  xs, ys = self.getXYData()
277  self.path_line.set_data(xs, ys)
278  self.path_points.set_data(xs, ys)
279 
280  if self.pathData.getSize() == self.maxPoints:
281  self.setState(STATE_ADJUST_POINTS)
282 
283  self.ax.figure.canvas.draw()
284 
285  elif self.drawing == STATE_ADJUST_POINTS: # Points moving state
286  self.dragIndex = None
287  for i, point in enumerate(self.pathData):
288  x = point.getX()
289  y = point.getY()
290  if sqrt((ex - x)**2 + (ey - y)**2) < self.tolerance:
291  self.dragIndex = i
292  break
