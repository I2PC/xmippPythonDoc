339

353  meanA = mtd.getValue(MDL_VOLUME_SCORE2,1)

356  self.mean = Float(mean)
357  self._store(self)

359  self._store(self)

361 
362  #statistic from fnal pdb with fsc-q
363  #Number of atoms greater or less than 0.5
364  total_atom=0
365  fscq_greater=0
366  fscq_less=0
367  with open(self._getFileName(PDB_VALUE_FILE)) as f:
368  lines_data = f.readlines()
369  for j,lin in enumerate(lines_data):
370 
371  if ( lin.startswith('ATOM') or lin.startswith('HETATM') ):
372 
373  total_atom=total_atom+1
374  fscq_atom = float(lin[54:60])
375 
376  if (fscq_atom>0.5):
377  fscq_greater=fscq_greater+1
378 
379  if (fscq_atom<-0.5):
380  fscq_less=fscq_less+1
381 
382  porc_greater=(fscq_greater*100)/total_atom
383  porc_less=(fscq_less*100)/total_atom
384

388  self._store(self)
