103  def _insertNewPartsSteps(self, insertedDict, inputParticles):
104  deps = []
105  writeSetOfParticles([m.clone() for m in inputParticles],
106  self._getExtraPath("allDone.xmd"),
107  alignType=ALIGN_NONE)
108  writeSetOfParticles([m.clone() for m in inputParticles
109  if int(m.getObjId()) not in insertedDict],
110  self._getExtraPath("newDone.xmd"),

112 
113  stepId = \
114  self._insertFunctionStep('kmeansClassifyStep',
115  self._getExtraPath("newDone.xmd"),
116  prerequisites=[])
117  deps.append(stepId)

120  insertedDict[part.getObjId()] = stepId

149  def _checkNewOutput(self):
150  if getattr(self, 'finished', False):
151  return
152  # Load previously done items (from text file)
153  doneList = self._readDoneList()
154  # Check for newly done items
155  partsSet = self._createSetOfParticles()
156  readSetOfParticles(self._getExtraPath("allDone.xmd"), partsSet)
157  newDone = [m.clone() for m in self.SetOfParticles
158  if int(m.getObjId()) not in doneList]
159  self.finished = self.streamClosed and (len(doneList) == len(partsSet))
160  if newDone:
161  self._writeDoneList(newDone)
162  elif not self.finished:
163  # If we are not finished and no new output have been produced
164  # it does not make sense to proceed and updated the outputs
165  # so we exit from the function here
166  return
167  if self.finished: # Unlock createOutputStep if finished all jobs
168  outputStep = self._getFirstJoinStep()
169  if outputStep and outputStep.isWaiting():
170  outputStep.setStatus(cons.STATUS_NEW)
