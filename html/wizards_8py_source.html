98  def getAutodownsampling(self,protocol, coeff=1.5):
99  samplingRate = protocol.inputMicrographs.get().getSamplingRate()
100  downsampling_factor = coeff / samplingRate
101  if downsampling_factor < 1:
102  downsampling_factor = 1
103  return downsampling_factor
104 
105  def show(self, form):
106  protocol = form.protocol
107  params = self._getParameters(protocol)
108  _value = params['value']
109  _label = params['label']
110 
111  if protocol.AutoDownsampling:
112  downSampling = self.getAutodownsampling(protocol)
113  else:
114  downSampling = _value[0]
115 
116 # form.setParamFromVar('inputMicrographs') # update selected input micrographs
117  provider = self._getProvider(protocol)
118 
119  if provider is not None:
120  args = {'unit': UNIT_PIXEL,
121  'downsample': downSampling,
122  'lf': _value[1],
123  'hf': _value[2],
124  'showInAngstroms': True
125  }
126  oldDowsample = downSampling
127  d = CtfDownsampleDialog(form.root, provider, **args)
128 
129  if d.resultYes():
130  newDownsample=d.getDownsample()
131  form.setVar(_label[0], newDownsample)
132  form.setVar(_label[1], d.getLowFreq())
133  form.setVar(_label[2], d.getHighFreq())
134  if abs(newDownsample-oldDowsample)>1e-4:
135  form.setVar("AutoDownsampling",False)
136  else:
137  dialog.showWarning("Empty input", "Select elements first", form.root)
