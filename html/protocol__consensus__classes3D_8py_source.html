451  def _objectiveFunction(self, ns, indices, vs, eps=0.01):
452  """Objective function to minimize based on the entropy and the similarity between the clusters to merge
453  """
454  p = self._calculateDistribution(ns)
455  pi = self._calculateDistribution(self._ensembleIntersectionPair(ns, indices))
456  h = self._calculateEntropy(p)
457  hi = self._calculateEntropy(pi)
458 
459  sv = self._calculateCosineSimilarity(vs[indices[0]], vs[indices[1]])
460 
461  return (h - hi) / (sv + eps)

463  def _ensembleIntersectionPair(self, intersections, indices):
464  """ Return a list of subsets where the intersections[indices] have been merged """
465  # Divide the input intersections into the ones referred by indices and not
466  selection = [intersections[i] for i in indices] # intersection[indices]
467  result = [intersections[i] for i in range(len(intersections)) if i not in indices] # intersections - selection
468

470  merged = XmippProtConsensusClasses3D.ParticleCluster.union(*selection)

472  # Add the merged items to the result

476  def _ensembleCheapestIntersectionPair(self, allClasses, intersections):
477  """ Merges the lest costly pair of intersections """
478  # Compute the similarity vectors
479  similarityVectors = list(map(self._buildSimilarityVector, intersections, itertools.repeat(allClasses)))
480 
481  # For each possible pair of intersections compute the cost of merging them
482  obValues = {}
483  for pair in itertools.combinations(range(len(intersections)), r=2):
484  obValues[pair] = self._objectiveFunction(intersections, pair, similarityVectors)
485 
486  values = list(obValues.values())
487  keys = list(obValues.keys())
488 
489  # Select the minimum cost pair and merge them
490  obValue = min(values)
491  mergePair = keys[values.index(obValue)]
492  mergedIntersections = self._ensembleIntersectionPair(intersections, mergePair)
493 
494  return mergedIntersections, obValue

519  def _calculateGaussianMleEstimates(self, d, q):
520  """ MLE estimates of guassian distributions
521  with common variance parameter
522  """
523  p = len(d)
524  d1 = d[:q]
525  d2 = d[q:]
526  mu1 = np.mean(d1)
527  mu2 = np.mean(d2)
528  var1 = np.var(d1)
529  var2 = np.var(d2)
530  sigma = ((q-1)*var1+(p-q-1)*var2)/(p-2)
531  theta1 = [mu1, sigma]
532  theta2 = [mu2, sigma]
533  return theta1, theta2

535  def _calculateGaussianDistribution(self, d, theta):
536  """ Gaussian pdf """
537  m = theta[0]
538  s = theta[1]
539  gauss = 1/(np.sqrt(2*np.pi*s))*np.exp(-1/(2*s)*(d-m)**2)

624  def _calculateReferenceClassification(self, classifications, numExec, threadPool=None):
625  """ Create random partitions of same size to compare the quality
626  of the classification """
627  clusterLengths = self._getClassificationLengths(classifications)
628  numParticles = sum(clusterLengths[0])
629 
630  # Repeatedly obtain a consensus of a random classification of same size
631  if threadPool:
632  consensus = threadPool.starmap(
633  self._calculateRandomClassificationConsensus,
634  itertools.repeat((clusterLengths, numParticles), numExec)
635  )
636  else:
637  consensus = list(map(
638  self._calculateRandomClassificationConsensus,
639  itertools.repeat(clusterLengths, numExec),
640  itertools.repeat(numParticles, numExec)
641  ))
642  assert(len(consensus) == numExec)
643

645  consensus = list(itertools.chain(*consensus))
