254  def classifyOneGroup(self, projNumber, projMdBlock, projRef,
255  mdClasses, mdImages):
256  """ Classify one of the neighbourhood groups if not empty.
257  Class information will be stored in output metadata: mdOut
258  """
259  blockSize = md.getSize(projMdBlock)
260  Nclasses = self.directionalClasses.get()
261  Nlevels = int(math.ceil(math.log(Nclasses) / math.log(2)))
262 
263  # Skip projection directions with not enough images to
264  # create a given number of classes
265  if blockSize / Nclasses < 10:
266  return
267 
268  fnDir = self._getExtraPath("direction_%s" % projNumber)
269  makePath(fnDir)
270 
271  # Run CL2D classification for the images assigned to one direction
272  args = "-i %s " % projMdBlock
273  args += "--odir %s " % fnDir
274  args += "--ref0 %s --iter %d --nref %d " % (
275  projRef, self.cl2dIterations, Nclasses)
276  args += "--distance correlation --classicalMultiref "
277  args += "--maxShift %f " % self.maxShift
278  try:
279  self.runJob("xmipp_classify_CL2D", args)
280  except:
281  return
282 
283  # After CL2D the stk and xmd files should be produced
284  classesXmd = join(fnDir, "level_%02d/class_classes.xmd" % Nlevels)
285  classesStk = join(fnDir, "level_%02d/class_classes.stk" % Nlevels)
286 
287  # Let's check that the output was produced
288  if not exists(classesStk):
289  return
290 
291  # Run align of the class average and the projection representative
292  fnAlignRoot = join(fnDir, "classes")
293  args = "-i %s " % classesStk
294  args += "--ref %s " % projRef
295  args += " --oroot %s --iter 1" % fnAlignRoot
296  self.runJob("xmipp_image_align", args, numberOfMpi=1)
297 
298  # Apply alignment
299  args = "-i %s_alignment.xmd --apply_transform" % fnAlignRoot
300  self.runJob("xmipp_transform_geometry", args, numberOfMpi=1)
301 
302  for classNo in range(1, Nclasses + 1):
303  localImagesMd = emlib.MetaData("class%06d_images@%s"
304  % (classNo, classesXmd))
305 
306  # New class detected
307  self.classCount += 1
308  # Check which images have not been assigned yet to any class
309  # and assign them to this new class
310  for objId in localImagesMd:
311  imgId = localImagesMd.getValue(emlib.MDL_ITEM_ID, objId)
312  # Add images not classify yet and store their class number
313  if imgId not in self.classImages:
314  self.classImages.add(imgId)
315  newObjId = mdImages.addObject()
316  mdImages.setValue(emlib.MDL_ITEM_ID, imgId, newObjId)
317  mdImages.setValue(emlib.MDL_REF2, self.classCount,
318  newObjId)
319 
320  newClassId = mdClasses.addObject()
321  mdClasses.setValue(emlib.MDL_REF, projNumber, newClassId)
322  mdClasses.setValue(emlib.MDL_REF2, self.classCount, newClassId)
323  mdClasses.setValue(emlib.MDL_IMAGE, "%d@%s" %
324  (classNo, classesStk), newClassId)
325  mdClasses.setValue(emlib.MDL_IMAGE1, projRef, newClassId)
326  mdClasses.setValue(emlib.MDL_CLASS_COUNT, localImagesMd.size(),
327  newClassId)
