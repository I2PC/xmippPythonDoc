37 def crop(x, output_shape, shifts_):
38  """
39  Extract from image 'x' all possible crops consistent with 'output_shape' and 'shifts'
40  :param x: an np.array with ndim = 2 representing a grayscale image
41  :param output_shape: A list / tuple / np.array with ndim = 2 representing [nb_rows, nb_cols]
42  :param shifts: A list / tuple / np.array with ndim = 2 representing the shifts in x and y axes for cropping the
43  image ([x_shift, y_shift])
44  :return: A list of np.arrays representing all possible crops consistent with the arguments. The np.arrays do not
45  include the channels information and therefore they have ndim = 2.
46  """
47  assert len(shifts_) == 2, "Invalid shift lengths"
48  assert shifts_[0] > 0 and shifts_[1] > 0, "Negative shifts"
49  # copy input argument
50  shifts = [shifts_[0], shifts_[1]]
51 
52  nb_segments_i = np.floor(x.shape[0] / float(output_shape[0]))
53  nb_segments_j = np.floor(x.shape[1] / float(output_shape[1]))
54  if nb_segments_i > 1:
55  delta_i_max = int(np.floor((x.shape[0] - output_shape[0]) / float(nb_segments_i - 1.0)))
56  if shifts[0] > delta_i_max:
57  shifts[0] = delta_i_max
58  else:
59  shifts[0] = 0
60  if nb_segments_j > 1:
61  delta_j_max = int(np.floor((x.shape[1] - output_shape[1]) / float(nb_segments_j - 1.0)))
62  if shifts[1] > delta_j_max:
63  shifts[1] = delta_j_max
64  else:
65  shifts[1] = 0
66  start_i = 0
67  end_i = start_i + output_shape[0]
68  Xs = []
69  for i in range(int(nb_segments_i)):
70  start_j = 0
71  end_j = start_j + output_shape[1]
72  for j in range(int(nb_segments_j)):
73  assert end_i <= x.shape[0], "end_i >= x_shape[0]"
74  assert end_j <= x.shape[1], "end_j >= x_shape[1]"
75  Xs.append(x[start_i:end_i, start_j:end_j])
76  start_j += shifts[1]
77  end_j = start_j + output_shape[1]
78  start_i += shifts[0]
79  end_i = start_i + output_shape[0]
80  return Xs
