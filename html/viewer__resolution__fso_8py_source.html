262  def interpolRes(self, thr, x, y):
263  """
264  This function is called by _showAnisotropyCurve.
265  It provides the cut point of the curve defined
266  by the points (x,y) with a threshold thr.
267  The flag okToPlot shows if there is no intersection points
268  """
269  idx = np.arange(0, len(x))
270  aux = np.array(y) <= thr
271  idx_x = idx[aux]
272  okToPlot = True
273  resInterp = []
274  if not idx_x.any():
275  okToPlot = False
276  else:
277  if len(idx_x) > 1:
278  idx_2 = idx_x[0]
279  idx_1 = idx_2 - 1
280  if idx_1 < 0:
281  idx_2 = idx_x[1]
282  idx_1 = idx_2 - 1
283  y2 = x[idx_2]
284  y1 = x[idx_1]
285  x2 = y[idx_2]
286  x1 = y[idx_1]
287  slope = (y2 - y1) / (x2 - x1)
288  ny = y2 - slope * x2
289  resInterp = 1.0 / (slope * thr + ny)
290  else:
291  okToPlot = False
292 
293  return resInterp, okToPlot
