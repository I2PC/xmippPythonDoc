310  @staticmethod
311  def colorMapToColorList(steps, colorMap):
312  """ Returns a list of pairs resolution, hexColor to be used in chimera
313  scripts for coloring the volume and the colorKey """
314 
315  # Get the map used by monoRes
316  colors = ()
317  ratio = 255.0 / (len(steps) - 1)
318  for index, step in enumerate(steps):
319  colorPosition = int(round(index * ratio))
320  rgb = colorMap(colorPosition)[:3]
321  colors += step,
322  rgbColor = mcolors.rgb2hex(rgb)
323  colors += rgbColor,
324 
325  return colors
