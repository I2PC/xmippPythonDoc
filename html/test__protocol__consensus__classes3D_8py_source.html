85  setOfClasses.append(newClass)

87  enabledClass = setOfClasses[newClass.getObjId()]
88  enabledClass.enableAppend()
89  for itemId in currIds:
90  item = partSet[itemId]
91  enabledClass.append(item)
92 
93  setOfClasses.update(enabledClass)
94 
95  # including the outputClasses as protocol output by hand
96  pImpClasses._defineOutputs(outputClasses=setOfClasses)
97 
98  # relaunching the protocol to get the new output
99  cls.proj.launchProtocol(pImpClasses, wait=True)
100
