19 def precalculateBands(nBand, dim, sampling, maxRes, minRes):
20 
21  vectorFreq = torch.fft.fftfreq(dim)
22  freq_band = torch.full((dim,int(np.floor(dim/2))), 50)
23 
24  maxFreq = sampling/maxRes
25  minFreq = sampling/minRes
26  factor = nBand * (1/maxFreq)
27 
28  for x in range(dim):
29 
30  if vectorFreq[x] >= 0:
31  wx = vectorFreq[x]
32 
33  for y in range(dim):
34  wy = vectorFreq[y]
35 
36  w = torch.sqrt(wx**2 + wy**2)
37 
38  if (w > minFreq) and (w < maxFreq):
39  freq_band[y][x] = torch.floor(w*factor)
40
