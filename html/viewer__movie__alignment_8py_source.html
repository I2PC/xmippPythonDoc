88  def _visualizeAny(self, outNameCondition):
89  views = []
90  # to show micrographs is prior than movies
91  objs = [x for x in self.protocol.iterOutputAttributes(SetOfMicrographs)]
92  objs += [x for x in self.protocol.iterOutputAttributes(SetOfMovies)]
93  for outName, outObj in objs:
94  if outNameCondition(outName):
95  views.append(self.objectView(outObj, viewParams=getViewParams()))
96  break
97  if not views:
98  outputType = 'discarded' if outNameCondition('Discarded') else 'accepted'
99  self.infoMessage('%s does not have %s outputs%s'
100  % (self.protocol.getObjLabel(), outputType,
101  getStringIfActive(self.protocol)),
102  title='Info message').show()
103  return views
