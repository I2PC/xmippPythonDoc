100  @classmethod
101  def readZernikeParams(cls, filename):
102  with open(filename, 'r') as fid:
103  lines = fid.readlines()
104  basis_params = np.fromstring(lines[0].strip('\n'), sep=' ')
105  coeffs = np.fromstring(lines[1].strip('\n'), sep=' ')
106  size = int(coeffs.size / 3)
107  coeffs = np.asarray([coeffs[:size], coeffs[size:2 * size], coeffs[2 * size:]])
108  return [int(basis_params[0]), int(basis_params[1]), basis_params[2], coeffs]

128  def testPairAlignmentZernike3DCPU(self):
129  # Test for CPU implementation
130  alignZernike = self.newProtocol(XmippProtVolumeDeformZernike3D,
131  inputVolume=self.protImportVol_1720.outputVolume,
132  refVolume=self.protImportVol_1723.outputVolume,
133  l1=3,
134  l2=2,
135  targetResolution=6,
136  useGpu=False,
137  numberOfThreads=4,
138  objLabel="Pair Alignment Zernike3D - CPU")
139  self.launchProtocol(alignZernike)
140  self.assertIsNotNone(alignZernike.outputVolume,
141  "There was a problem with XmippProtVolumeDeformZernike3D - CPU Version")
142  # Check coefficients and parameters are the ones expected
143  clnm_gold = self.readZernikeParams(self.clnm_pd_cpu_gold)
144  clnm_test = self.readZernikeParams(alignZernike._getExtraPath('Volumes_clnm.txt'))
145  self.assertEqual(clnm_gold[0], clnm_test[0], "There is a problem with Zernike degree")
146  self.assertEqual(clnm_gold[1], clnm_test[1], "There is a problem with Spherical Harmonics degree")
147  self.assertEqual(clnm_gold[2], clnm_test[2], "There is a problem with sphere radius")
148  diff_clnm = np.abs(np.sum(clnm_gold[3] - clnm_test[3]))
149  self.assertAlmostEqual(diff_clnm, 0, delta=0.2,
150  msg="Zernike coefficients do not have the expected value")
