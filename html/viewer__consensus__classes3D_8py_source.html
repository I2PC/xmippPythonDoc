230  def _plotDendrogram(self, fig, ax, clusterings, obValues):
231  """ Plot the dendrogram from the objective functions of the merge
232  between the groups of images """
233 
234  # Initialize required values
235  linkageMatrix = np.zeros((len(clusterings)-1, 4))
236  clsIds = np.arange(len(clusterings))
237  nIntersections = len(clusterings[0])
238 
239  # Loop over each iteration of clustering
240  for i in range(len(clusterings)-1):
241  # Find the sets that were merged
242  clsMerged = []
243  assert(len(clusterings[i]) == len(clsIds))
244  for cls, clsId in zip(clusterings[i], clsIds):
245  if cls not in clusterings[i+1]:
246  clsMerged.append(clsId)
247  assert(len(clsMerged) == 2)
248  clsMerged = np.array(clsMerged)
249 
250  # Find original number of sets within new set
251  nCls = list(map(lambda x : linkageMatrix[x,3] if (x >= 0) else 1, clsMerged - nIntersections))
252 
253  # Create linkage matrix
254  linkageMatrix[i, 0:2] = clsMerged # ids of merged sets
255  linkageMatrix[i, 2] = obValues[i+1] # objective function as distance
256  linkageMatrix[i, 3] = sum(nCls) # total number of original sets
257 
258  # Change set ids to reflect new set of clusters
259  for id in clsMerged:
260  clsIds = np.delete(clsIds, np.argwhere(clsIds == id))
261  clsIds = np.append(clsIds, len(clusterings)+i)
262 
263  # Plot resulting dendrogram
264  hierarchy.dendrogram(linkageMatrix, ax=ax)
265  ax.set_title('Dendrogram')
266  ax.set_xlabel('sets ids')
267  ax.set_ylabel('objective function')
268  ax.set_ylim([np.min(linkageMatrix[:, 2]), np.max(linkageMatrix[:, 2])])
