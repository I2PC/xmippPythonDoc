26 def _mean_centered_pca(samples: torch.Tensor,
27  k: int) -> Tuple[torch.Tensor, torch.Tensor]:
28  if len(samples.shape) != 2:
29  raise RuntimeError('Sample matrix is expected to have 2 dimensions')
30 
31  if len(samples) >= 3*k:
32  # Perform the computations with the smallest
33  # covariance vector possible
34  transpose = samples.shape[0] < samples.shape[1]
35 
36  # Compute the covariance according to
37  # the transposition
38  if transpose:
39  covariance = samples @ samples.T
40  else:
41  covariance = samples.T @ samples
42  covariance /= len(samples) - 1
43  assert(covariance.shape == (min(samples.shape), )*2)
44 
45  # Compute the largest eigenvalues of the covariance matrix
46  eigenvalues, eigenvectors = torch.lobpcg(
47  covariance,
48  k=k
49  )
50 
51  # Undo the transposition
52  if transpose:
53  eigenvectors = samples.T @ eigenvectors
54  eigenvectors /= torch.norm(eigenvectors, dim=0, keepdim=True)
55 
56  else:
57  eigenvalues = torch.zeros((k, ), dtype=samples.dtype, device=samples.device)
58  eigenvectors = torch.zeros((samples.shape[-1], k), dtype=samples.dtype, device=samples.device)
59 
60  return eigenvalues, eigenvectors

63 def aligned_2d_classification(dataset: Iterable[torch.Tensor],
64  scratch: torch.Tensor ):
65  # Write
66  start = 0
67  for vectors in dataset:
68  end = start + len(vectors)
69 
70  # Write
71  scratch[start:end,:] = vectors.to(scratch, non_blocking=True)
72 
73  # Setup next iteration
74  start = end
75 
76  # Perform the PCA analysis
77  avg = scratch.mean(dim=0)
78  scratch -= avg
79  _, v = _mean_centered_pca(scratch, k=1)
80  direction = v[:,0]
81 
82 
83  projections = torch.matmul(scratch, direction[...,None])[:,0]
84 
85  return avg, direction, projections
