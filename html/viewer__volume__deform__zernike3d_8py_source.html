96  def _doShowStrain(self, param=None):
97 
98  scriptFile = self.protocol._getPath('strain_chimera.cxc')
99  fhCmd = open(scriptFile, 'w')
100  fnbase = removeExt(self.protocol._getFileName('fnInputVol'))

113  fhCmd.write(self.VOXEL_SIZE % (counter, str(smprt)))

115  fhCmd.write('color sample #%d map #%d palette rainbow\n' % (counter - 1, counter))
116  fhCmd.write(self.VIEW)
