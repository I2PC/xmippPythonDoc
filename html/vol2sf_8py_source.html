69  Makes use of the Hermitian symmetry to speed up.

85  List
86  """
87  if r > vol.shape[0]//2 or r > vol.shape[1]//2 or r > vol.shape[2]//2:

91  else:
92  pass

96 
97  # only half
98  the = np.pi*(2*np.arange(b)+1) / (4*b)
99  phi = np.pi*np.arange(2*b) / b
100 
101  # C order mesh
102  phi, the = np.meshgrid(phi, the)
103  the = the.flatten()
104  phi = phi.flatten()
105 
106  # compute the coordinates
107  x = r*np.cos(phi)*np.sin(the) + m_x
108  y = r*np.sin(phi)*np.sin(the) + m_y
109  z = r*np.cos(the) + m_z
110 
111  # use spline intepolation on the real/imaginary parts
112  # can be done better, but now it suffices
113  vol_r = np.real(vol)
114  vol_i = np.imag(vol)
115 
116  from scipy.ndimage import map_coordinates
117  res_r_a = map_coordinates(vol_r, [x, y, z], order=2)
118  res_i_a = map_coordinates(vol_i, [x, y, z], order=2)
119 
120  # fill in the other half
121  ind = np.arange(2*b**2)
122  cind = (2*b-1-(ind+2*b**2)//(2*b))*2*b + np.mod(np.mod(ind, 2*b)+b, 2*b)
123 
124  res_r = np.zeros((4*b**2,))
125  res_r[:2*b**2] = res_r_a
126  res_r[2*b**2:] = res_r_a[cind]
127 
128  res_i = np.zeros((4*b**2,))
129  res_i[:2*b**2] = res_i_a
130  res_i[2*b**2:] = -res_i_a[cind]
131 
132  return [res_r, res_i]

136  """Do a Fourier shift of a spherical function to save some interpolation time.
137 
138  Parameters

141  numpy.ndarray

144  Integer

152  dy: Shift in Y dim in real space
153  Integer / float
154 
155  dz: Shift in Z dim in real space
156  Integer / float
157

159  Integer
160 
161  Returns
162  -------
163  A shifted (complex) spherical function in Fourier space.
164  """
165  b = int(len(sf)**0.5/2)
166  assert len(sf) == 4*b**2
167 
168  the = np.pi*(2*np.arange(2*b)+1) / (4*b)
169  phi = np.pi*np.arange(2*b) / b
170 
171  # C order mesh
172  phi, the = np.meshgrid(phi, the)
173  the = the.flatten()
174  phi = phi.flatten()
175 
176  # compute the coordinates
177  xx = r*np.cos(phi)*np.sin(the)
178  yy = r*np.sin(phi)*np.sin(the)
179  zz = r*np.cos(the)
180 
181  sx = shape[0]
182  sy = shape[1]
183  sz = shape[2]
184 
185  shift = np.exp(-2j*np.pi/sx*xx*dx) * np.exp(-2j*np.pi/sy*yy*dy) * np.exp(-2j*np.pi/sz*zz*dz)
186 
187  return sf*shift
