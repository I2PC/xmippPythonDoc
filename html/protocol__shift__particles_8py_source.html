92  def shiftStep(self):
93  """call xmipp program to shift the particles"""
94  centermd = self._getExtraPath("center_particles.xmd")
95  args = '-i "%s" -o "%s" ' % (self._getExtraPath("input_particles.xmd"), centermd)
96  if self.option:
97  self.x = self.xin.get()

99  self.z = self.zin.get()
100  else:
101  fnvol = self.inputMask.get().getFileName()
102  if fnvol.endswith('.mrc'):
103  fnvol += ':mrc'
104  vol = Volume()
105  vol.setFileName(fnvol)
106  vol = ih().read(vol.getFileName())
107  masscenter = vol.centerOfMass()
108  self.x = masscenter[0]
109  self.y = masscenter[1]
110  self.z = masscenter[2]
111 
112  args += '--shift_to %f %f %f ' % (self.x, self.y, self.z)
113  program = "xmipp_transform_geometry"
114  if not self.interp.get():
115  interp = 'linear'
116  else:
117  interp = 'spline'
118  if self.applyShift.get():
119  args += ' --apply_transform'
120  args += ' --dont_wrap --interp %s' % interp
121  if self.inv.get():
122  args += ' --inverse'
123  self.runJob(program, args)
124
