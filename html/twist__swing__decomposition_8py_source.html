29 def twist_decomposition(quaternions: torch.Tensor,
30  direction: Union[torch.Tensor, int],
31  assume_normalized: bool = True,
32  normalize_output: bool = True,
33  out: Optional[torch.Tensor] = None) -> torch.Tensor:
34 
35  if not assume_normalized:
36  raise NotImplementedError()
37 
38  # Create the output
39  out = torch.empty(quaternions.shape, dtype=quaternions.dtype, device=quaternions.device, out=out)
40 
41  if isinstance(direction, int):
42  # Only copy the selected axis to te output
43  for i in range(3):
44  out[...,i+1] = quaternions[...,i+1] if i == direction else 0
45 
46  else:
47  # Compute the dot product of the direction and quaternion xyz
48  # and store it on the w component of the output
49  torch.matmul(direction[...,None,:], quaternions[...,1:4,None], out=out[...,0,None])
50 
51  # Compute the xyz components by scaling the direction
52  # with the dot product. This is equivalent to the
53  # projection of q.xyz onto the direction
54  torch.mul(direction, out[...,0,None], out=out[1:4])
55 
56  # Overwrite the w component with the w component of the
57  # input quaternions
58  out[...,0] = quaternions[...,0]
59 
60  # Normalize
61  if normalize_output:
62  out /= torch.norm(out, dim=-1, keepdim=True)
63 
64  return out
