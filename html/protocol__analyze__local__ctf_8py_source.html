64  def analyzeDefocus(self):
65  """compute R2 coefficient of each micrograph and prepare data to be later displayed in the viewer as a 3D
66  representation of the distribution of particles in the micrograph"""
67  micIds=[]
68  particleIds=[]
69  x=[]
70  y=[]
71  defocusU=[]
72  defocusV=[]
73  for particle in self.inputSet.get():
74  micIds.append(particle.getMicId())
75  particleIds.append(particle.getObjId())
76  xi, yi = particle.getCoordinate().getPosition()
77  x.append(xi)
78  y.append(yi)
79  defocusU.append(particle.getCTF().getDefocusU())
80  defocusV.append(particle.getCTF().getDefocusV())
81 
82  uniqueMicIds = list(set(micIds))
83  self.R2={}
84 
85  md = emlib.MetaData()
86 
87  for micId in uniqueMicIds:
88  idx = [i for i,j in enumerate(micIds) if j==micId]
89  defocusUbyId = []
90  defocusVbyId = []
91  meanDefocusbyId = []
92  xbyId = []
93  ybyId = []
94  particleIdsbyMicId = []
95  for idxi in idx:
96  defocusUbyId.append(defocusU[idxi])
97  defocusVbyId.append(defocusV[idxi])
98  meanDefocus = (defocusU[idxi]+defocusV[idxi])/2
99  meanDefocusbyId.append(meanDefocus)
100  xbyId.append(x[idxi])
101  ybyId.append(y[idxi])
102  particleIdsbyMicId.append(particleIds[idxi])
103 
104  #defocus = c*y + b*x + a = A * X; A=[x(i),y(i)]
105  A = np.column_stack([np.ones(len(xbyId)),xbyId,ybyId])
106  polynomial, residuals, _, _ = np.linalg.lstsq(A,meanDefocusbyId,rcond=None)
107  meanDefocusbyIdArray = np.asarray(meanDefocusbyId)
108  coefficients = np.asarray(polynomial)
109  self.R2[micId] = 1 - residuals / sum((meanDefocusbyIdArray - meanDefocusbyIdArray.mean()) ** 2)
110  mdBlock = emlib.MetaData()
111  for xi, yi, deltafi, parti in zip(xbyId,ybyId,meanDefocusbyId,particleIdsbyMicId):
112  objId = mdBlock.addObject()
113  mdBlock.setValue(emlib.MDL_ITEM_ID,long(parti),objId)
114  mdBlock.setValue(emlib.MDL_XCOOR,xi,objId)
115  mdBlock.setValue(emlib.MDL_YCOOR,yi,objId)
116  mdBlock.setValue(emlib.MDL_CTF_DEFOCUSA,deltafi,objId)
117  estimatedVal = coefficients[2]*yi + coefficients[1]*xi + coefficients[0]
118  residuali = deltafi - estimatedVal
119  mdBlock.setValue(emlib.MDL_CTF_DEFOCUS_RESIDUAL,residuali,objId)
120  mdBlock.write("mic_%d@%s"%(micId,self._getExtraPath("micrographDefoci.xmd")),emlib.MD_APPEND)
121  objId = md.addObject()
122  md.setValue(emlib.MDL_CTF_DEFOCUS_COEFS,coefficients.tolist(),objId)
123  md.write(self._getExtraPath("micrographCoef.xmd"),emlib.MD_APPEND)
