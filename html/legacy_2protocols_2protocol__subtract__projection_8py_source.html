121  def convertInputStep(self, volName):
122  # Read volume and convert to DOUBLE
123  self.vol = emlib.Image(volName)
124  self.vol.convert2DataType(emlib.DT_DOUBLE)
125  # Mask volume if needed
126  if self.refMask.get() is not None:
127  maskName = getImageLocation(self.refMask.get())
128  self.mask = emlib.Image(maskName)
129  self.mask.convert2DataType(emlib.DT_DOUBLE)
130  self.vol.inplaceMultiply(self.mask)
131  padding = 2
132  maxFreq = 0.5
133  splineDegree = 3
134 
135  self.fourierProjectVol = emlib.FourierProjector(self.vol, padding, maxFreq, splineDegree)
136 
137  partSet = self.inputParticles.get()
138  nPart = len(partSet)
139  numberOfTasks = min(nPart, max(self.numberOfThreads.get()-1, 1))
140  taskSize = int(nPart / numberOfTasks)
141  md = emlib.MetaData()
142 
143  # Convert angles and shifts from volume system of coordinates to
144  # projection system of coordinates
145  mdCount = 0
146 
147  for index, part in enumerate(partSet):
148  objId = md.addObject()
149  imgRow = emlib.metadata.Row()
150  particleToRow(part, imgRow)
151  shifts, angles = geometryFromMatrix(part.getTransform().getMatrix(), True)
152 
153  imgRow.setValue(emlib.MDL_SHIFT_X,-shifts[0])
154  imgRow.setValue(emlib.MDL_SHIFT_Y,-shifts[1])
155  imgRow.setValue(emlib.MDL_SHIFT_Z,0.)
156  imgRow.setValue(emlib.MDL_ANGLE_ROT,angles[0])
157  imgRow.setValue(emlib.MDL_ANGLE_TILT,angles[1])
158  imgRow.setValue(emlib.MDL_ANGLE_PSI,angles[2])
159 
160  imgRow.writeToMd(md, objId)
161 
162  # Write a new metadata every taskSize number of elements
163  # except in the last chunk where we want to add also the
164  # remainder and the condition is the last element
165  if ((index % taskSize == taskSize-1 and
166  mdCount < numberOfTasks-1) or index == nPart-1):
167  md.write(self._getInputParticlesSubsetFn(mdCount))
168  md.clear()
169  mdCount += 1
170

172  emlib.createEmptyFile(self._getProjGalleryFn(), x, y, 1, nPart)

174  def projectStep(self, start, end, samplingRate, threadNumber):
175  # Project
176  md = emlib.MetaData(self._getInputParticlesSubsetFn(threadNumber))
177 
178  projection = emlib.Image()
179  projection.setDataType(emlib.DT_DOUBLE)
180 
181  for id in md:
182  rot = md.getValue(emlib.MDL_ANGLE_ROT, id)
183  tilt = md.getValue(emlib.MDL_ANGLE_TILT, id)
184  psi = md.getValue(emlib.MDL_ANGLE_PSI, id)
185 
186 
187  self.fourierProjectVol.projectVolume(projection, rot, tilt, psi)
188 
190  if self.projType == self.CORRECT_NONE:
191  pass
192  elif self.projType == self.CORRECT_FULL_CTF:
193  emlib.applyCTF(projection, md, samplingRate, id, False)
194  elif self.projType == self.CORRECT_PHASE_FLIP:
195  emlib.applyCTF(projection, md, samplingRate, id, True)
196  else:
197  raise Exception("ERROR: Unknown projection mode: %d" % self.projType)
198 
199  # Shift image
200  projection.applyGeo(md,id,True,False)#onlyapplyshist, wrap
201  ih = ImageHandler()
202  expProj = ih.read(md.getValue(emlib.MDL_IMAGE, id))
203  expProj.convert2DataType(emlib.DT_DOUBLE)
204  # Subtract from experimental and write result
205  projection.resetOrigin()
206  if self.normalize:
207  expProj = expProj.adjustAndSubtract(projection)
208  else:
209  expProj.inplaceSubtract(projection)
210 
211  expProj.write( self._getProjGalleryIndexFn(id+start-1))
