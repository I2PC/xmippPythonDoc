42 class XmippProtMovieMaxShift(ProtProcessMovies):
43  """
44  Protocol to make an automatic rejection of those movies whose
45  frames move more than a given threshold.
46  Rejection criteria:
47  - *by frame*: Rejects movies with drifts between frames
48  bigger than a certain maximum.
49  - *by whole movie*: Rejects movies with a total travel
50  bigger than a certain maximum.
51  - *by frame and movie*: Rejects movies if both conditions
52  above are met.
53  - *by frame or movie*: Rejects movies if one of the conditions
54  above are met.
55  """

120  def _evaluateMovieAlign(self, movieId):
121  """ Fill the accepted or the rejected list with the movie.
122  """
123  movie = self.inputMovies.get()[movieId].clone()
124  alignment = movie.getAlignment()
125  sampling = self.samplingRate
126 
127  # getShifts() returns the absolute shifts from a certain reference
128  shiftListX, shiftListY = alignment.getShifts()
129 
130  # initialize the criteria values
131  rejectedByMovie = False
132  rejectedByFrame = False
133  if any(shiftListX) or any(shiftListY):
134  # we use np.arrays to use np.diff()
135  shiftArrayX = np.asarray(shiftListX)
136  shiftArrayY = np.asarray(shiftListY)
137 
138  evalBoth = self.rejType==self.REJ_AND or self.rejType==self.REJ_OR
139  if self.rejType == self.REJ_MOVIE or evalBoth:
140  rangeX = np.max(shiftArrayX) - np.min(shiftArrayX)
141  rangeY = np.max(shiftArrayY) - np.min(shiftArrayY)
142  rangeM = max(rangeX, rangeY) * sampling
143  rejectedByMovie = rangeM > self.maxMovieShift.get()
144 
145  if self.rejType == self.REJ_FRAME or evalBoth:
146  frameShiftX = np.abs(np.diff(shiftArrayX))
147  frameShiftY = np.abs(np.diff(shiftArrayY))
148  maxShiftX = np.max(frameShiftX)
149  maxShiftY = np.max(frameShiftY)
150  maxShiftM = max(maxShiftX, maxShiftY) * sampling
151  rejectedByFrame = maxShiftM > self.maxFrameShift.get()
152 
153  if self.rejType == self.REJ_AND:
154  if rejectedByFrame and rejectedByMovie:
155  self.discardedMoviesList.append(movie)
156  else:
157  self.acceptedMoviesList.append(movie)
158  else: # for the OR and the individuals evaluations
159  if rejectedByFrame or rejectedByMovie:
160  self.discardedMoviesList.append(movie)
161  else:
162  self.acceptedMoviesList.append(movie)
163  else: # we accept the movie if no shifts is associated
164  self.acceptedMoviesList.append(movie)
