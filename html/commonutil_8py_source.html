72

75  """

78  if len(ty) != len(pv):
79  raise ValueError("len(ty) must be equal to len(pv)")

84  sumy = ty.sum()

130  assert lower < upper
131  l, n = x.shape

133  feat_max = x.max(axis=0).toarray().flatten()
134  coef = (feat_max - feat_min) / (upper - lower)
135  coef[coef != 0] = 1.0 / coef[coef != 0]
136 
137  # (x - ones(l,1) * feat_min') * diag(coef) + lower
138  # = x * diag(coef) - ones(l, 1) * (feat_min' * diag(coef)) + lower
