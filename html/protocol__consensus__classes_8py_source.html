266  def _calculateIntersections(self,
267  classifications: Iterable[Iterable[Set[int]]],
268  ids: Iterable[Iterable[int]] ) -> Sequence[Set[int]]:
269  # Start with the first classification
270  itCls = iter(classifications)
271  itId = iter(ids)
272 
273  result = next(itCls)
274  traces = list(map('001.{:d}'.format, next(itId)))
275 
276  for i, (classification, ids) in enumerate(zip(itCls, itId), start=2):
277  intersections = []
278  intersection_traces = []
279 
280  # Perform the intersection with previous classes
281  for cls0, trace in zip(result, traces):
282  for cls1, id in zip(classification, ids):
283  intersection = cls0.intersection(cls1)
284  if len(intersection) > 0:
285  intersections.append(intersection)
286  intersection_traces.append(trace + ' & {:03d}.{:d}'.format(i, id))
287 
288  # Use the new intersection for the next step
289  result = intersections
290  traces = intersection_traces
291 
292  assert(len(result) == len(traces))
293  return result, traces

354  def _calculateLinkage(self,
355  intersections: List[Set[int]],
356  classes: List[Set[int]],
357  metric: str ) -> np.ndarray:
358 
359  linkage = np.zeros((len(intersections)-1, 4))
360 
361  # Initialize the working data structures
362  similarities = list(map(lambda x : self._calculateClusterSimilarityVector(x, classes), intersections))
363  clusters = list(enumerate(intersections))
364 
365  for id, row in enumerate(linkage, start=len(intersections)):
366  # Determine the indices to be merged
367  # TODO optimize to not use squareform
368  distances = scipy.spatial.distance.pdist(similarities, metric=metric)
369  distance_matrix = scipy.spatial.distance.squareform(distances, 'tomatrix')
370  np.fill_diagonal(distance_matrix, np.inf)
371  idx0, idx1 = np.unravel_index(np.argmin(distance_matrix), distance_matrix.shape)
372 
373  # Be careful to pop from the back first
374  idx0, idx1 = max(idx0, idx1), min(idx0, idx1)
375 
376  # Merge clusters
377  row[0], cluster0 = clusters.pop(idx0)
378  row[1], cluster1 = clusters.pop(idx1)
379  row[2] = distance_matrix[idx0, idx1]
380  # TODO row[3] is not written
381  merged = cluster0.union(cluster1)
382  clusters.append((id, merged))
383 
384  # Add the new similarity
385  similarities.pop(idx0)
386  similarities.pop(idx1)
387  similarities.append(self._calculateClusterSimilarityVector(merged, classes))
388 
389  assert(len(similarities) == len(clusters))
390 
391  assert(len(similarities) == 1)
392  assert(len(clusters) == 1)
393  return linkage

395  def _calculateMergedIntersections(self,
396  intersections: List[Set[int]],
397  linkage: np.ndarray,
398  stop: int = 1):
399  merged = [intersections]
400  clusters = intersections.copy()
401 
402  for idx0, idx1, _, _ in linkage:
403  # Construct a new cluster joining two of the clusters
404  cluster0: Set[int] = clusters[int(idx0)]
405  cluster1: Set[int] = clusters[int(idx1)]
406  clusterMerged = cluster0.union(cluster1)
407  clusters.append(clusterMerged)
408 
409  # Copy the previous row except for the
410  # merged sets. Add the new merged set
411  assert(cluster0 in merged[-1])
412  assert(cluster1 in merged[-1])
413  mergedNew = [clusterMerged]
414  for cluster in merged[-1]:
415  if cluster not in (cluster0, cluster1):
416  mergedNew.append(cluster)
417 
418  merged.append(mergedNew)
419 
420  # Stop
421  if len(mergedNew) <= stop:
422  break
423 
424  return merged

426  def _calculatePercentile(self, data: np.ndarray, value: float):
427  """ Given an array of values (data), finds the corresponding percentile of the value.
428  Percentile is returned in range [0, 1] """
429  assert(np.array_equal(sorted(data), data)) # In order to iterate it in ascending order

459  def _calculateElbowProfileLikelihood(self, cost: np.ndarray) -> int:
460  def cost_fun(i: int) -> float:
461  # Partition the function in q
462  d1 = cost[:i]
463  d2 = cost[i:]
464 
465  # Compute the average and mean of each partition
466  mu1 = np.mean(d1)
467  mu2 = np.mean(d2)
468  sigma1 = np.std(d1, ddof=1)
469  sigma2 = np.std(d2, ddof=1)
470 
471  # Compute the log likelihood
472  logLikelihoods1 = scipy.stats.norm.logpdf(d1, mu1, sigma1)
473  logLikelihoods2 = scipy.stats.norm.logpdf(d2, mu2, sigma2)
474  return np.sum(logLikelihoods1) + np.sum(logLikelihoods2)
475 
476  if len(cost) > 4:
477  return max(range(2, len(cost)-1), key=cost_fun)
478  else:
479  return len(cost) // 2

488  def _calculateElbowDistanceToLine(self, cost: np.ndarray) -> int:
489  # Based on:
490  # https://en.wikipedia.org/wiki/Distance_from_a_point_to_a_line
491  a = cost - cost[0]
492  b = np.linspace(0.0, cost[-1] - cost[0], len(cost))
