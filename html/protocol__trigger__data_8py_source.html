39 class XmippProtTriggerData(EMProtocol):
40  """
41  Waits until certain number of images is prepared and then
42  send them to output.
43  It can be done in 3 ways:
44  - If *Send all particles to output?*' is _No_:
45  Once the number of images is reached, a setOfImages is returned and
46  the protocols finished (ending the streaming from this point).
47  - If *Send all particles to output?*' is _Yes_ and:
48  - If *Split particles to multiple sets?* is _Yes_:
49  Multiple closed outputs will be returned as soon as
50  the number of images is reached.
51  - If *Split particles to multiple sets?* is _No_:
52  Only one output is returned and it is growing up in batches of
53  a certain number of images (completely in streaming).
54  """
55  _label = 'trigger data'
56  _lastUpdateVersion = VERSION_2_0

108  def _checkNewInput(self):
109  imsFile = self.inputImages.get().getFileName()
110  self.lastCheck = getattr(self, 'lastCheck', datetime.now())
111  mTime = datetime.fromtimestamp(os.path.getmtime(imsFile))
112 
113  # If the input's sqlite have not changed since our last check,
114  # it does not make sense to check for new input data
115  if self.lastCheck > mTime and hasattr(self, 'newImages'):
116  return None
117 
118  # loading the input set in a dynamic way
119  inputClass = self.getImagesClass()
120  self.imsSet = inputClass(filename=imsFile)
121  self.imsSet.loadAllProperties()
122 
123  # loading new images to process
124  if len(self.images) > 0: # taking the non-processed yet
125  self.newImages = [m.clone() for m in self.imsSet.iterItems(
126  orderBy='creation',
127  where='creation>"' + str(self.check) + '"')]
128  else: # first time
129  self.newImages = [m.clone() for m in self.imsSet]

132  self.images = self.images + self.newImages
133  if len(self.newImages) > 0:

135  direction='DESC'):
