463  def rejectByPopulation(self, ids):
464  if self.usePopulation.get() and self.classesDict is not None:
465  sizeDict = {clsId: size for clsId, size
466  in self.classesDict.items()
467  if clsId in ids}
468 
469  meanPop = sum(sizeDict.values())/len(sizeDict)
470 
471  for clsId, size in sizeDict.items():
472  # minPopulation is normalized to 100% not to 1
473  decision = int(size*100 > meanPop * self.minPopulation.get())
474  self.enableCls[clsId] = ACCEPTED if decision else DISCARDED
475  else:
476  self.enableCls = {clsId: ACCEPTED for clsId in ids}
