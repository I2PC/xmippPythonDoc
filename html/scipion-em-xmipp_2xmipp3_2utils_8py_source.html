79 def rotation(imag, angle, shape, P):
80  '''Rotate a np.array and return also the transformation matrix
81  #imag: np.array
82  #angle: angle in degrees
83  #shape: output shape
84  #P: transform matrix (further transformation in addition to the rotation)'''
85  (hsrc, wsrc) = imag.shape
86  angle *= math.pi / 180
87  T = np.asarray([[1, 0, -wsrc / 2], [0, 1, -hsrc / 2], [0, 0, 1]])
88  R = np.asarray([[math.cos(angle), math.sin(angle), 0], [-math.sin(angle), math.cos(angle), 0], [0, 0, 1]])
89  M = np.matmul(np.matmul(np.linalg.inv(T), np.matmul(R, T)), P)
90 
91  transformed = applyTransform(imag, M, shape)
92  return transformed, M

95 def flipYImage(inFn, outFn=None, outDir=None):
96  '''Flips an image in the Y axis'''
97  if outFn == None:
98  if not '_flipped' in basename(inFn):
99  ext = pwutils.getExt(inFn)
100  outFn = inFn.replace(ext, '_flipped' + ext)
101  else:
102  outFn = inFn.replace('_flipped', '')
103  if outDir != None:
104  outFn = outDir + '/' + basename(outFn)
105  if isfile(outFn):
106  return outFn
107  gainImg = readImage(inFn)
108  imag_array = np.asarray(gainImg.getData(), dtype=np.float64)
109 
110  # Flipped Y matrix
111  M, angle = np.asarray([[1, 0, 0], [0, -1, imag_array.shape[0]], [0, 0, 1]]), 0
112  flipped_array, M = rotation(imag_array, angle, imag_array.shape, M)
113  writeImageFromArray(flipped_array, outFn)

145 def surrounding_values(a, ii, jj, depth=1):
146  '''Return a list with the surrounding elements, given the indexs of the center, from an 2D numpy array
147  '''
148  values = []
149  for i in range(ii - depth, ii + depth + 1):
150  for j in range(jj - depth, jj + depth + 1):
151  if i >= 0 and j >= 0 and i < a.shape[0] and j < a.shape[1]:
152  if i != ii or j != jj:
153  values += [a[i][j]]
154  return values

286  def splitLongestSegment(self):
287  """ Split the longest segment by adding the midpoint. """
288  maxDist = 0
289  n = self.getSize()
290  # Find the longest segment and its index
291  for i in range(n - 1):
292  p1 = self.getPoint(i)
293  x1, y1 = p1.getX(), p1.getY()
294  p2 = self.getPoint(i + 1)
295  x2, y2 = p2.getX(), p2.getY()
296  dist = (x1 - x2) ** 2 + (y1 - y2) ** 2
297  if dist > maxDist:
298  maxDist = dist
299  maxIndex = i + 1
300  midX = (x1 + x2) / 2
301  midY = (y1 + y2) / 2
302  # Add a midpoint to it
303  point = self.createEmptyPoint()
304  point.setX(midX)
305  point.setY(midY)
306  self.addPoint(point, position=maxIndex)
