27 def euler_to_matrix(rot: torch.Tensor,
28  tilt: torch.Tensor,
29  psi: torch.Tensor,
30  out: Optional[torch.Tensor] = None) -> torch.Tensor:
31 
32  # Create the output
33  batch_shape = torch.broadcast_shapes(rot.shape, tilt.shape, psi.shape)
34  result_shape = batch_shape + (3, 3)
35  dtype = rot.dtype
36  device = rot.device
37  out = torch.empty(result_shape, dtype=dtype, device=device, out=out)
38 
39  ai = -psi
40  aj = -tilt
41  ak = -rot
42 
43  # Obtain sin and cos of the half angles
44  ci = torch.cos(ai)
45  si = torch.sin(ai)
46  cj = torch.cos(aj)
47  sj = torch.sin(aj)
48  ck = torch.cos(ak)
49  sk = torch.sin(ak)
50 
51  # Obtain the combinations
52  ci_ck = ci * ck
53  ci_sk = ci * sk
54  si_ck = si * ck
55  si_sk = si * sk
56 
57  # Build the matrix
58  out[...,0,0] = cj * ci_ck - si_sk
59  out[...,0,1] = cj * si_ck + ci_sk
60  out[...,0,2] = -sj * ck
61  out[...,1,0] = -cj * ci_sk - si_ck
62  out[...,1,1] = -cj * si_sk + ci_ck
63  out[...,1,2] = sj * sk
64  out[...,2,0] = sj * ci
65  out[...,2,1] = sj * si
66  out[...,2,2] = cj
67 
68  return out
