142  def estimatePoissonCount(self, movie):
143  mean_frames = []
144  n = movie.getNumberOfFrames()
145  frames = [1, n/2, n]
146 
147  for frame in frames:
148  frame_image = ImageHandler().read("%d@%s" % (frame, movie.getFileName())).getData()
149  mean_dose_per_pixel = np.mean(frame_image)
150  mean_dose_per_angstrom2 = mean_dose_per_pixel/ self.samplingRate**2
151  mean_frames.append(mean_dose_per_angstrom2)
152 
153  stats = computeStats(np.asarray(mean_frames))
154  self.meanDoseList.append(stats['mean'])
155 
156  return stats

178  def _checkNewOutput(self):
