208  def createMaskFromVolumeStep(self):
209  volume = self.inputVolume.get()
210  fnVol = getImageLocation(volume)
211  Ts = volume.getSamplingRate()
212 
213  if self.volumeOperation == OPERATION_THRESHOLD:
214  self.runJob("xmipp_transform_threshold",
215  "-i %s -o %s --select below %f --substitute binarize"
216  % (fnVol, self.maskFile, self.threshold.get()))
217  elif self.volumeOperation == OPERATION_SEGMENT:
218  args="-i %s -o %s --method " % (fnVol, self.maskFile)
219  if self.segmentationType == SEGMENTATION_VOXELS:
220  args += "voxel_mass %d" % (self.nvoxels.get())
221  elif self.segmentationType == SEGMENTATION_AMINOACIDS:
222  args += "aa_mass %d %f" % (self.naminoacids.get(), Ts)
223  elif self.segmentationType == SEGMENTATION_DALTON:
224  args += "dalton_mass %d %f" % (self.dalton.get(), Ts)
225  else:
226  args += "otsu"
227  self.runJob("xmipp_volume_segment", args)
228 
229  elif self.volumeOperation == OPERATION_POSTPROCESS:
230  ImageHandler().convert(fnVol,self.maskFile)
231 
232  return [self.maskFile]

265  if self.symmetry!='c1':
266  self.runJob("xmipp_transform_symmetrize","-i %s --sym %s --dont_wrap"%(self.maskFile,self.symmetry.get()))
267  if self.volumeOperation==OPERATION_THRESHOLD or self.volumeOperation==OPERATION_SEGMENT:
268  self.runJob("xmipp_transform_threshold",
269  "-i %s --select below 0.5 --substitute binarize" % self.maskFile)
270 
271  if self.doMorphological:
272  self.runJob("xmipp_transform_morphology","-i %s --binaryOperation %s --size %d"
273  % (self.maskFile, self.getEnumText('morphologicalOperation'),
274  self.elementSize.get()))
275 
276  if self.doInvert:
