52  def quaternion_from_matrix(matrix, isprecise=False):
53  """Return quaternion from rotation matrix.
54 
55  If isprecise is True, the input matrix is assumed to be a precise rotation
56  matrix and a faster algorithm is used."""
57  M = np.array(matrix, dtype=np.float64, copy=False)[:4, :4]
58  if isprecise:
59  q = np.empty((4,))
60  t = np.trace(M)
61  if t > M[3, 3]:
62  q[0] = t
63  q[3] = M[1, 0] - M[0, 1]
64  q[2] = M[0, 2] - M[2, 0]
65  q[1] = M[2, 1] - M[1, 2]
66  else:
67  i, j, k = 1, 2, 3
68  if M[1, 1] > M[0, 0]:
69  i, j, k = 2, 3, 1
70  if M[2, 2] > M[i, i]:
71  i, j, k = 3, 1, 2
72  t = M[i, i] - (M[j, j] + M[k, k]) + M[3, 3]
73  q[i] = t
74  q[j] = M[i, j] + M[j, i]
75  q[k] = M[k, i] + M[i, k]
76  q[3] = M[k, j] - M[j, k]
77  q *= 0.5 / math.sqrt(t * M[3, 3])
78  else:
79  m00 = M[0, 0]
80  m01 = M[0, 1]
81  m02 = M[0, 2]
82  m10 = M[1, 0]
83  m11 = M[1, 1]
84  m12 = M[1, 2]
85  m20 = M[2, 0]
86  m21 = M[2, 1]
87  m22 = M[2, 2]
88  # symmetric matrix K
89  K = np.array([[m00 - m11 - m22, 0.0, 0.0, 0.0],
90  [m01 + m10, m11 - m00 - m22, 0.0, 0.0],
91  [m02 + m20, m12 + m21, m22 - m00 - m11, 0.0],
92  [m21 - m12, m02 - m20, m10 - m01, m00 + m11 + m22]])
93  K /= 3.0
94  # quaternion is eigenvector of K that corresponds to largest eigenvalue
95  w, V = np.linalg.eigh(K)
96  q = V[[3, 0, 1, 2], np.argmax(w)]
97  if q[0] < 0.0:
98  np.negative(q, q)
99  return q

102  def quaternion_matrix(quaternion):
103  """Return homogeneous rotation matrix from quaternion."""
104  q = np.array(quaternion, dtype=np.float64, copy=True)
105  n = np.dot(q, q)
106  if n < _EPS:
107  return np.identity(4)
108  q *= math.sqrt(2.0 / n)
109  q = np.outer(q, q)
110  return np.array([
111  [1.0 - q[2, 2] - q[3, 3], q[1, 2] - q[3, 0], q[1, 3] + q[2, 0], 0.0],
112  [q[1, 2] + q[3, 0], 1.0 - q[1, 1] - q[3, 3], q[2, 3] - q[1, 0], 0.0],
113  [q[1, 3] - q[2, 0], q[2, 3] + q[1, 0], 1.0 - q[1, 1] - q[2, 2], 0.0],
114  [0.0, 0.0, 0.0, 1.0]])

117  def euler_from_matrix(matrix, axes='sxyz'):
118  """Return Euler angles from rotation matrix for specified axis sequence.
119 
120  axes : One of 24 axis sequences as string or encoded tuple
121 
122  Note that many Euler angle triplets can describe one matrix."""
123  try:
124  firstaxis, parity, repetition, frame = _AXES2TUPLE[axes.lower()]
125  except (AttributeError, KeyError):
126  _TUPLE2AXES[axes]
127  firstaxis, parity, repetition, frame = axes
128 
129  i = firstaxis
130  j = _NEXT_AXIS[i + parity]
131  k = _NEXT_AXIS[i - parity + 1]
132 
133  M = np.array(matrix, dtype=np.float64, copy=False)[:3, :3]
134  if repetition:
135  sy = math.sqrt(M[i, j] * M[i, j] + M[i, k] * M[i, k])
136  if sy > _EPS:
137  ax = math.atan2(M[i, j], M[i, k])
138  ay = math.atan2(sy, M[i, i])
139  az = math.atan2(M[j, i], -M[k, i])
140  else:
141  ax = math.atan2(-M[j, k], M[j, j])
142  ay = math.atan2(sy, M[i, i])
143  az = 0.0
144  else:
145  cy = math.sqrt(M[i, i] * M[i, i] + M[j, i] * M[j, i])
146  if cy > _EPS:
147  ax = math.atan2(M[k, j], M[k, k])
148  ay = math.atan2(-M[k, i], cy)
149  az = math.atan2(M[j, i], M[i, i])
150  else:
151  ax = math.atan2(-M[j, k], M[j, j])
152  ay = math.atan2(-M[k, i], cy)
153  az = 0.0
154 
155  if parity:
156  ax, ay, az = -ax, -ay, -az
157  if frame:
158  ax, az = az, ax
159  return ax, ay, az

167  def rotation6d_to_matrix(rot):
168  """Return rotation matrix from 6D representation."""
169  a1 = np.array((rot[0], rot[2], rot[4]))
170  a2 = np.array((rot[1], rot[3], rot[5]))
171  a1 = np.reshape(a1, (3, 1))
172  a2 = np.reshape(a2, (3, 1))
173 
174  b1 = a1 / np.linalg.norm(a1)
175 
176  c1 = np.multiply(b1, a2)
177  c2 = np.sum(c1)
178 
179  b2 = a2 - c2 * b1
180  b2 = b2 / np.linalg.norm(b2)
181  b3 = np.cross(b1, b2, axis=0)
182  return np.concatenate((b1, b2, b3), axis=1)
