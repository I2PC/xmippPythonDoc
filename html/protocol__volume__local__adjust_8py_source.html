60  def adjustStep(self):

64  if fnVol1.endswith('.mrc'):

66  if vol2.endswith('.mrc'):

68  program = "xmipp_local_volume_adjust"
69  args = '--i1 %s --i2 %s -o %s --mask %s --neighborhood %d --sampling %s' % \
70  (fnVol1, vol2, self._getExtraPath("output_volume.mrc"), self.mask.get().getFileName(),
71  self.neighborhood.get(), vol1.getSamplingRate())
72  if self.subtract.get():
