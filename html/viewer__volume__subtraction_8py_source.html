107  _outputVol.getFileName()))
108 
109  # output vol origin coordinates
110  x_output, y_output, z_output = self.protocol.outputVolume.getShiftsFromOrigin()
111  f.write("open %s\n" % outputVolFileName)
112  f.write("volume #2 voxelSize %f origin "
113  "%0.2f,%0.2f,%0.2f\n"

115  f.close()
