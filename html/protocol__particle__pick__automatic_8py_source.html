136  def _pickMicrograph(self, mic, *args):
137  micPath = mic.getFileName()
138  # Get particle picking boxsize from the previous run
139  modelRoot = self._getExtraPath('model')
140 
141  micName = removeBaseExt(micPath)
142  proceed = True
143  if self.micsToPick == MICS_SAMEASPICKING:
144  basePos = replaceBaseExt(micPath, "pos")
145  fnPos = self.particlePickingRun._getExtraPath(basePos)
146  if exists(fnPos):
147  blocks = emlib.getBlocksInMetaDataFile(fnPos)
148  copy = True
149  if 'header' in blocks:
150  mdheader = emlib.MetaData("header@" + fnPos)
151  state = mdheader.getValue(emlib.MDL_PICKING_MICROGRAPH_STATE,
152  mdheader.firstObject())
153  if state == "Available":
154  copy = False
155  if copy:
156  # Copy manual .pos file of this micrograph
157  copyFile(fnPos, self._getExtraPath(basename(fnPos)))
158  proceed = False
159 
160  if proceed:
161  args = "-i %s " % micPath
162  args += "--particleSize %d " % self.boxSize
163  args += "--model %s " % modelRoot
164  args += "--outputRoot %s " % self._getExtraPath(micName)
165  args += "--mode autoselect --thr %d" % self.numberOfThreads
166 
167  self.runJob("xmipp_micrograph_automatic_picking", args)

176  def _validate(self):
177  validateMsgs = []
178 
179  if self.modelSource == SRC_MANUAL_PICKING and not hasattr(self.xmippParticlePicking.get(),"outputCoordinates"):
180  validateMsgs.append("You need to generate coordinates for the "
181  "supervised picking")
182 
183  srcDir = self.getSrcDir()
184  srcPaths = [os.path.join(srcDir,k) for k in self.filesToCopy]
185  # Check that all needed files exist
186  if missingPaths(*srcPaths):
187  validateMsgs.append('Input picking run has not been trained, '
188  'use *Autopick* for at least one micrograph')
189 
190  # If other set of micrographs is provided they should have same
191  # sampling rate and acquisition
192  if self.micsToPick.get() == MICS_OTHER and self.modelSource != SRC_DIR:
193  inputMics = self.inputMicrographs.get()
194  manualMics = self.xmippParticlePicking.get().inputMicrographs.get()
195  # FIXME: manualMics is always None when scheduled...
196  # it should be fixed in the update step at Scipion scheduler app
197  if manualMics is not None:
198  pixsizeInput = inputMics.getSamplingRate()
199 
200  pixsizeMics = manualMics.getSamplingRate()
201  acq = manualMics.getAcquisition()
202 
203  if pixsizeInput != pixsizeMics:
204  validateMsgs.append('New micrographs should have same sampling '
205  'rate as the ones already picked.')
206 
207  if not inputMics.getAcquisition().equalAttributes(acq):
208  validateMsgs.append('New micrographs should have same '
209  'acquisition parameters as the ones '
210  'already picked.')
211 
212  if self.modelSource.get()==SRC_DIR and self.micsToPick.get()==MICS_SAMEASPICKING:
213  validateMsgs.append("You cannot take the model from a directory and indicate that the set of micrograohs "
214  "is the same as picking. If you take the model from a directory, probably you want "
