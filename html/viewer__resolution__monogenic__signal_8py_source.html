374  def colorMapToColorList(steps, colorMap):
375  """ Returns a list of pairs resolution, hexColor to be used in chimera
376  scripts for coloring the volume and the colorKey """
377 
378  # Get the map used by monoRes
379  colors = ()
380  ratio = 255.0/(len(steps)-1)
381  for index, step in enumerate(steps):
382  colorPosition = int(round(index*ratio))
383  rgb = colorMap(colorPosition)[:3]
384  colors += step,
385  rgbColor = mcolors.rgb2hex(rgb)
386  colors += rgbColor,
387 
388  return colors
