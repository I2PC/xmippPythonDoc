55  def computeVolumeConsensus(self, inputFile, outVolFn, wavelet='sym11'):
56  outputWt = None
57  outputMin = None
58  nlevel = 3
59  fnCoef = splitext(outVolFn)[0] + '_coef.txt'
60  fhCoef = open(fnCoef, 'w')
61  with open(inputFile) as f:
62  for line in f:
63  fileName = line.split()[0]
64  if fileName.endswith('.mrc'):
65  fileName += ':mrc'
66  V = xmippLib.Image(line.split()[0])
67  vol = V.getData()
68  wt = pywt.swtn(vol, wavelet, nlevel)
69  if outputWt == None:
70  outputWt = wt
71  outputMin = wt[0]['aaa']*0
72  else:
73  for level in range(0, nlevel):
74  wtLevel = wt[level]
75  outputWtLevel = outputWt[level]
76  for key in wtLevel:
77  outputWtLevel[key] = np.where(np.abs(outputWtLevel[key]) > np.abs(wtLevel[key]),
78  outputWtLevel[key], wtLevel[key])
79  fhCoef.write(str(outputWtLevel[key]))
80  diff = np.abs(np.abs(outputWtLevel[key]) - np.abs(wtLevel[key]))
81  outputMin = np.where(outputMin > diff, outputMin, diff)
82  f.close()
83  fhCoef.close()
84  consensus = pywt.iswtn(outputWt, wavelet)
85  V = xmippLib.Image()
86  V.setData(consensus)
87  V.write(outVolFn)
88  V.setData(outputMin)
89  outVolFn2 = splitext(outVolFn)[0] + '_diff.mrc'
