144  def convertPdbStep(self):
145  vol1 = self.vol1.get()
146  pdbFn = self._getPdbFileName()
147  self.outFile = self._getVolName()
148  if getExt(pdbFn) == ".cif":
149  pdbFn2 = replaceBaseExt(pdbFn, 'pdb')
150  cifToPdb(pdbFn, pdbFn2)
151  pdbFn = pdbFn2
152  samplingR = vol1.getSamplingRate()
153  size = vol1.getDim()
154  ccp4header = headers.Ccp4Header(vol1.getFileName(), readHeader=True)
155  self.shifts = ccp4header.getOrigin()
156  # convert pdb to volume with the size and origin of the input volume
157  args = ' -i %s --sampling %f -o %s --size %d %d %d --orig %d %d %d' % \
158  (pdbFn, samplingR, removeExt(self.outFile), size[2], size[1], size[0], self.shifts[0]/samplingR,
159  self.shifts[1]/samplingR, self.shifts[2]/samplingR)
160  program = "xmipp_volume_from_pdb"
161  self.runJob(program, args)
162  volpdbmrc = "%s.mrc" % removeExt(self.outFile)
163  # convert volume from pdb to .mrc in order to store the origin in the mrc header
164  args2 = ' -i %s -o %s -t vol' % (self.outFile, volpdbmrc)
165  program2 = "xmipp_image_convert"
166  self.runJob(program2, args2)
167  # write origin in mrc header of volume from pdb
168  ccp4headerOut = headers.Ccp4Header(volpdbmrc, readHeader=True)
169  ccp4headerOut.setOrigin(self.shifts)
