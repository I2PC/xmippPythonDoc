37 class XmippProtSubtractProjection(EMProtocol):
38  """ This protocol computes the subtraction between particles and a initial volume, by computing its projections
39  with the same angles that input particles have. Then, each particle and the correspondent projection of the initial
40  volume are numerically adjusted and subtracted using a mask which denotes the region to keep. """

42  _label = 'subtract projection'
