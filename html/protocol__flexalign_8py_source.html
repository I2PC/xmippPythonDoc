175  def getUserFlip(self, imag_array):
176  flipDic = {0: np.asarray([[1, 0, 0], [0, 1, 0], [0, 0, 1]]),
177  1: np.asarray([[1, 0, 0], [0, -1, imag_array.shape[0]], [0, 0, 1]]),
178  2: np.asarray([[-1, 0, imag_array.shape[1]], [0, 1, 0], [0, 0, 1]])}
179  return flipDic[self.gainFlip.get()]

266  def transformGain(self, gainFn, outFn=None):
267  '''Transforms the gain image with the user especifications'''
268  if outFn == None:
269  ext = pwutils.getExt(gainFn)
270  baseName = os.path.basename(gainFn).replace(ext, '_transformed' + ext)
271  outFn = os.path.abspath(self._getExtraPath(baseName))
272 
273  if os.path.isfile(outFn):
274  return outFn
275 
276  gainImg = xmutils.readImage(gainFn)
277  imag_array = np.asarray(gainImg.getData(), dtype=np.float64)
278 
279  # Building the transformation matrix
280  angle = self.getUserAngle()
281  M = self.getUserFlip(imag_array)
282  print('Transforming gain: {} degrees rotation, {} flip'.
283  format(angle, ['no', 'vertical', 'horizontal'][self.gainFlip.get()]))
284  flipped_array, M = xmutils.rotation(imag_array, angle, imag_array.shape, M)
285  xmutils.writeImageFromArray(flipped_array, outFn)
286  return outFn
