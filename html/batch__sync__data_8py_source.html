222 def md5sum(fname):
223  """ Return the md5 hash of file fname
224  """
225  mhash = hashlib.md5()
226  with open(fname, 'rb') as f:
227  for chunk in iter(lambda: f.read(128 * mhash.block_size), b""):
228  mhash.update(chunk)
229  return mhash.hexdigest()

231 def createMANIFEST(path):
232  """ Create a MANIFEST file in path with the md5 of all files below
233  """
234  with open(join(path, 'MANIFEST'), 'wb') as manifest:
235  for root, dirs, files in os.walk(path):
236  for filename in set(files) - {'MANIFEST'}: # all but ourselves
