156  def _readCoordinates(self, coordsDir, suffix=''):
157  micTiltPairs = self.inputMicrographsTiltedPair.get()
158  uSuffix = 'Untilted' + suffix
159  tSuffix = 'Tilted' + suffix
160  uSet = micTiltPairs.getUntilted()
161  tSet = micTiltPairs.getTilted()
162  # Create Untilted and Tilted SetOfCoordinates
163  uCoordSet = self._createSetOfCoordinates(uSet, suffix=uSuffix)
164  convert.readSetOfCoordinates(coordsDir, uSet, uCoordSet)
165  uCoordSet.write()
166  tCoordSet = self._createSetOfCoordinates(tSet, suffix=tSuffix)
167  convert.readSetOfCoordinates(coordsDir, tSet, tCoordSet)
168  tCoordSet.write()
169  boxSize = self._getBoxSize()
170  if boxSize:
171  uCoordSet.setBoxSize(boxSize)
172  tCoordSet.setBoxSize(boxSize)
173 
174  return uCoordSet, tCoordSet
