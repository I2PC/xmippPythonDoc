100 def consensusCoordsOneMic(coords_files, boxSize, consensusRadius, consensusCriterium, outDir):
101  """ Compute consensus of coordiantes for the same micrograph
102  @param coords_files [mic1_p1.pos, mic2_p2.pos,...]
103  """
104  baseName = os.path.splitext(os.path.basename(coords_files[0]))[0]
105  out_name= os.path.join(outDir, baseName+".pos")
106  if os.path.isfile(out_name): return
107  coords=[]
108  Ncoords=0
109  n=0
110  micIds=set([])
111  for fname in coords_files:
112  x_y_list, micId= readPosCoordsFromFName(fname, returnAlsoMicId=True)
113  if micId is None:
114  continue
115  micIds.add(micId)
116  x_y_array= np.asarray(x_y_list, dtype=int)
117  coords.append(x_y_array)
118  Ncoords += x_y_array.shape[0]
119  n += 1
120  if len(micIds)>1:
121  raise ValueError("Error, inconsistency in micIds, all files should belong to the same micId: %s"%(str(coords_files)))
122  if len(coords)==0:
123  return
124  allCoords = np.zeros([Ncoords, 2])
125  votes = np.zeros(Ncoords)
126 
127  # Add all coordinates of the first set of coordinates
128  N0 = coords[0].shape[0]
129  inAllMicrographs = consensusCriterium <= 0 or consensusCriterium == len(coords_files)
130  if N0 == 0 and inAllMicrographs:
131  return
132  elif N0 > 0:
133  allCoords[0:N0, :] = coords[0]
134  votes[0:N0] = 1
135 
136  consensusNpixels = consensusRadius* boxSize
137 
138  # Add the rest of coordinates
139  Ncurrent = N0
140 # for n in range(1, len(coords_files)):
141  for n in range(len(coords)):
142  for coord in coords[n]:
143  if Ncurrent > 0:
144  dist = np.sum((coord - allCoords[0:Ncurrent])**2, axis=1)
145  imin = np.argmin(dist)
146  if sqrt(dist[imin]) < consensusNpixels:
147  newCoord = (votes[imin]*allCoords[imin,]+coord)/(votes[imin]+1)
148  allCoords[imin,] = newCoord
149  votes[imin] += 1
150  else:
151  allCoords[Ncurrent, :] = coord
152  votes[Ncurrent] = 1
153  Ncurrent += 1
154  else:
155  allCoords[Ncurrent, :] = coord
156  votes[Ncurrent] = 1
157  Ncurrent += 1
158 
159  # Select those in the consensus
160  if consensusCriterium <= 0:
161  consensusCriterium = len(coords_files)
162 
163  consensusCoords = allCoords[votes >= consensusCriterium, :]
164  # Write the consensus file only if there are some coordinates (size > 0)
165  if consensusCoords.size>0:
166  writeCoordsListToPosFname(out_name, consensusCoords, outputRoot=outDir, micId= micId)
167

169
