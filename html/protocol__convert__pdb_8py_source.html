273  def _convertAtomStruct(self, pdbRaw):
274  """ This function receives an atomic struct file, and conditionally converts it to .pdb if it is in .cif format. """
275  # Get output path for atomic struct file
276  baseExt = replaceBaseExt(pdbRaw, 'pdb') if self.convertCif.get() else os.path.basename(pdbRaw)
277  convertedPdb = self._getExtraPath(baseExt).replace(" ", "_")
278 
279  # If file extension is .cif and conversion is required, convert to .pdb, or else we link it as is
280  if getExt(pdbRaw) == ".cif" and self.convertCif.get():
281  cifToPdb(pdbRaw, convertedPdb)
282  else:
283  createLink(pdbRaw, convertedPdb)
284 
285  # Return optionally converted file
286  return convertedPdb

288  def _getConversionArgs(self, pdb, samplingRate, outputFile, isSet=False):
289  """ This function generates the arguments to convert an input pdb to a volume. """
290  # Main arguments
291  args = '-i "%s" --sampling %f -o "%s"' % (pdb, samplingRate, outputFile)
292 
293  # Flag for centered pdbs
294  if self.centerPdb:
295  args += ' --centerPDB'
296  # Flag for output pdbs
297  if self.outPdb:
298  args += ' --oPDB'
299 
300  # Setting size and origin if selected
301  if isSet:
302  args += ' --size %d' % (self.size.get())
303  else:
304  if self.vol:
305  vol = self.volObj.get()
306  size = vol.getDim()
307  ccp4header = headers.Ccp4Header(vol.getFileName(), readHeader=True)
308  self.shifts = ccp4header.getOrigin()
309  args += ' --size %d %d %d --orig %d %d %d' % (size[2], size[1], size[0],
310  self.shifts[0]/samplingRate,
311  self.shifts[1]/samplingRate,
312  self.shifts[2]/samplingRate)
313  else:
314  if self.setSize:
315  args += ' --size'
316 
317  if self.size_x.hasValue():
318  args += ' %d' % self.size_x.get()
319 
320  if self.size_y.hasValue() and self.size_z.hasValue():
321  args += ' %d %d' % (self.size_y.get(), self.size_z.get())
322 
323  # Returning produced args
324  return args
