332  def calculateTiltCorrelationStep(self, mic):
333  psds = []
334  correlations = []
335  autocorrelations = []
336  # Read image
337  micFolder = self._getOutputMicFolder(mic)
338  micImage = ImageHandler().read(mic.getLocation())
339  dimx, dimy, z, n = micImage.getDimensions()
340  wind_step = self.window_size.get()
341  x_steps, y_steps = window_coordinates2D(dimx, dimy, wind_step)
342  subWindStep = int(wind_step * (self.samplingRate / self.objective_resolution.get()))
343  x_steps_psd, y_steps_psd = window_coordinates2D(subWindStep * 2, subWindStep * 2, subWindStep)
344  # Extract windows
345  window_image = ImageHandler().createImage()
346  rotatedWind_psd = ImageHandler().createImage()
347  output_image = ImageHandler().createImage()
348  output_array = np.zeros(((subWindStep * 2)-1, (subWindStep * 2)-1))
349  ih = ImageHandler()
350 
351  for i in range(0, 3, 2):
352  for j in range(0, 3, 2):
353  window = micImage.window2D(x_steps[i], y_steps[j], x_steps[i + 1], y_steps[j + 1])
354  x_dim, y_dim, z, n = window.getDimensions()
355  # NORMALIZED
356  mean, dev, minCorr, maxCorr = window.computeStats()
357  winMatrix = (window.getData() - mean) / dev
358  window_image.setData(winMatrix)
359  # Compute PSD
360  wind_psd = window_image.computePSD(0.4, x_dim, y_dim, 1)
361  wind_psd.convertPSD()
362  # Rotate PSD
363  psdMatrix = wind_psd.getData()
364  P = np.identity(3)
365  rotatedPSD_matrix, M = rotation(psdMatrix, 90, psdMatrix.shape, P)
366  rotatedWind_psd.setData(rotatedPSD_matrix)
367  # Window intro a sub window psd for the correlations
368  x0_sub = int((x_dim / 2) - (subWindStep / 2))
369  y0_sub = int((x_dim / 2) - (subWindStep / 2))
370 
371  subWind_psd = wind_psd.window2D(x0_sub, y0_sub, int(x0_sub + subWindStep - 1),
372  int(y0_sub + subWindStep - 1))
373 
374  subRotatedWind_psd = rotatedWind_psd.window2D(x0_sub, y0_sub, int(x0_sub + subWindStep - 1),
375  int(y0_sub + subWindStep - 1))
376  # SAVE images
377  filename = "tmp" + str(i) + str(j) + '.mrc'
378  window_image.write(os.path.join(micFolder, filename))
379  filename_subwindPSD = os.path.join(micFolder,
380  "tmp_psd" + str(i) + str(j) + '.mrc')
381  filename_rotatedPSD = os.path.join(micFolder,
382  "tmp_psd_rotated" + str(i) + str(j) + '.mrc')
383  subWind_psd.write(filename_subwindPSD)
384  subRotatedWind_psd.write(filename_rotatedPSD)
385  # Filter this window using runJob
386  filename_subwindPSD_filt = os.path.join(micFolder,
387  "tmp_psd_filtered" + str(i) + str(j) + '.mrc')
388  filename_subwindRotatedPSD_filt = os.path.join(micFolder,
389  "tmp_psd_rot_filtered" + str(i) + str(j) + '.mrc')
390 
391  args1 = '-i %s -o %s --fourier low_pass 0.1' % (filename_subwindPSD, filename_subwindPSD_filt)
392  args2 = '-i %s -o %s --fourier low_pass 0.1' % (filename_rotatedPSD, filename_subwindRotatedPSD_filt)
393 
394  self.runJob("xmipp_transform_filter", args1)
395  self.runJob("xmipp_transform_filter", args2)
396  # Calculate autocorrelation 90 degrees
397  subWind_psd_filt = ih.read(filename_subwindPSD_filt)
398  subRotatedWind_psd_filt = ih.read(filename_subwindRotatedPSD_filt)
399  autocorrelation = subWind_psd_filt.correlation(subRotatedWind_psd_filt)
400  # Paint the output array
401  output_array[y_steps_psd[j]:y_steps_psd[j] + subWindStep, x_steps_psd[i]:x_steps_psd[i] + subWindStep] = \
402  subWind_psd_filt.getData()
403  # Append
404  autocorrelations.append(autocorrelation)
405  psds.append(subWind_psd_filt)
406 
407  output_image.setData(output_array)
408  filename = "psd_outputs" + str(mic.getObjId()) + '.jpeg'
409  output_image.write(self._getExtraPath(filename))
410 
411  correlation_pairs = list(combinations(psds, 2))
412  for m1, m2 in correlation_pairs:
413  correlation = m1.correlation(m2)
414  correlations.append(correlation)
415 
416  correlations.extend(autocorrelations)
417  return correlations

468  inputFrame = os.path.join(micOrigin, line.strip())
