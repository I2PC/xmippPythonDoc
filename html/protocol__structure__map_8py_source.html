100  def convertStep(self, volFn, volDim, volSr, minDim, maxSr, nVoli):
101  Xdim = volDim
102  Ts = volSr
103  newTs = self.targetResolution.get() * 1.0 / 3.0
104  newTs = max(maxSr, newTs)
105  newXdim = Xdim * Ts / newTs
106  fnOut = os.path.splitext(volFn)[0]
107  fnOut = self._getExtraPath(os.path.basename(fnOut + self.OUTPUT_SUFFIX % nVoli))
108 
109  ih = ImageHandler()
110  if Xdim != newXdim:
111  self.runJob("xmipp_image_resize",
112  "-i %s -o %s --dim %d" % (volFn, fnOut, newXdim))
113 
114  else:
115  ih.convert(volFn, fnOut)
116 
117  if newXdim>minDim:
118  self.runJob("xmipp_transform_window", " -i %s -o %s --crop %d" %
119  (fnOut, fnOut, (newXdim - minDim)))

137  def computeCorr(self, vol1, vol2, i, j):
138  vol = xmippLib.Image(vol1)
139  defVol = xmippLib.Image(vol2)
140  corr = vol.correlation(defVol)
141  corr = 1 - corr
142  outFile = self._getExtraPath('Pair_%d_%d_correlation.txt' % (i,j))
143  with open(outFile, 'w') as f:
