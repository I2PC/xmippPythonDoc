107  def interpolRes(self, thr, x, y):
108  """
109  This function is called by _showAnisotropyCurve.
110  It provides the cut point of the curve defined
111  by the points (x,y) with a threshold thr.
112  The flag okToPlot shows if there is no intersection points
113  """
114  idx = np.arange(0, len(x))
115  aux = np.array(y) <= thr
116  idx_x = idx[aux]
117  okToPlot = True
118  resInterp = []
119  if not idx_x.any():
120  okToPlot = False
121  else:
122  if len(idx_x) > 1:
123  idx_2 = idx_x[0]
124  idx_1 = idx_2 - 1
125  if idx_1 < 0:
126  idx_2 = idx_x[1]
127  idx_1 = idx_2 - 1
128  y2 = x[idx_2]
129  y1 = x[idx_1]
130  x2 = y[idx_2]
131  x1 = y[idx_1]
132  slope = (y2 - y1) / (x2 - x1)
133  ny = y2 - slope * x2
134  resInterp = 1.0 / (slope * thr + ny)
135  else:
136  okToPlot = False
137 
138  return resInterp, okToPlot
