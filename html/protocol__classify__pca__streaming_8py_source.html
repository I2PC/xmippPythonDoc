85  # --------------------------- INSERT steps functions ----------------------
86  def stepsGeneratorStep(self) -> None:
87  """
88  This step should be implemented by any streaming protocol.
89  It should check its input and when ready conditions are met
90  call the self._insertFunctionStep method.
91  """
92  self._initialStep()
93  self.newDeps = []
94  newParticlesSet = self._loadEmptyParticleSet()
95 
96  if self.isContinued() and self._isPcaDone():
97  self.info('Continue protocol')
98  self._updateVarsToContinue()
99 
100  while not self.finish:
101  if not self._newParticlesToProcess():
102  self.info('No new particles')
103  else:
104  with self._lock: # This lock needs to be here since the classifyItems access the inputSet
105  # and can create a race condition
106  particlesSet = self._loadInputParticleSet()
107  self.streamState = particlesSet.getStreamState()
108  where = None
109  if self.lastCreationTime:
110  where = 'creation>"' + str(self.lastCreationTime) + '"'
111 
112  for particle in particlesSet.iterItems(orderBy='creation', direction='ASC', where=where):
113  tmp = particle.getObjCreation()
114  newParticlesSet.append(particle.clone())
115 
116  self.lastCreationTime = tmp
117  self.info('%d new particles' % len(newParticlesSet))
118  self.info('Last creation time REGISTER %s' % self.lastCreationTime)
119 
120  # ------------------------------------ PCA TRAINING ----------------------------------------------
121  if self._doPcaTraining(newParticlesSet):
122  self.pcaStep = self._insertFunctionStep(self.runPCASteps, newParticlesSet, prerequisites=[])
123  if len(newParticlesSet) == len(self.inputParticles.get()) and self.streamState == Set.STREAM_CLOSED:
124  self.staticRun = True
125  self.info('Static Run')
126  # ------------------------------------ CLASSIFICATION ----------------------------------------------
127  if self._doClassification(newParticlesSet):
128  self._insertClassificationSteps(newParticlesSet, self.lastCreationTime)
129  newParticlesSet = self._loadEmptyParticleSet()
130 
131  if self.streamState == Set.STREAM_CLOSED:
132  self.info('Stream closed')
133  # Finish everything and close output sets
134  if len(newParticlesSet):
135  self.info('Finish processing with last batch %d' % len(newParticlesSet))
136  self.lastRound = True
137  else:
138  self._insertFunctionStep(self.closeOutputStep, prerequisites=self.newDeps)
139  self.finish = True
140  continue # To avoid waiting 1 min
141 
142  sys.stdout.flush()
143  time.sleep(60)
144 
145  sys.stdout.flush() # One last flush

405  def _doPcaTraining(self, newParticlesSet):
406  """ Two cases for launching PCA steps:
407  - If there are enough particles and PCA has not been launched or finished.
408  - Launch if it's the last round of new particles and PCA has not been launched or finished. """
409  return (len(newParticlesSet) >= self.training.get() and not self._isPcaDone() and not self.pcaLaunch) or \
410  (self.lastRound and not self._isPcaDone() and not self.pcaLaunch)

412  def _doClassification(self, newParticlesSet):
413  """ Three cases for launching Classification
414  - First round of classification: enough particles and PCA done
415  - Update classification: classification done, PCA done and enough batch size
416  - First or update classification with a smaller batch: last round of particles and not classification launch
417  """
418  return (len(newParticlesSet) >= self.classificationBatch.get() and self._isPcaDone()
419  and not self._isClassificationDone() and not self.classificationLaunch) \
420  or (len(newParticlesSet) >= BATCH_UPDATE and self._isClassificationDone() and self._isPcaDone()
421  and not self.classificationLaunch) \
422  or (self.lastRound and not self.classificationLaunch)
